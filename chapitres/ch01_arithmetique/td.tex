%% ACTIVITE
\begin{Activite}[][10 min]
On considère la liste de nombres suivante :
$-3 \ ; \ 14 \ ; \ 2,5 \ ; \ 0 \ ; \ 25 \ ; \ 36 \ ; \ -7 \ ; \ 41$

\begin{enumerate}
    \item Parmi les nombres de cette liste, relever ceux qui sont des entiers positifs.

    (\textit{Ces nombres sont appelés des \textbf{entiers naturels}}).

    \item Parmi les entiers naturels trouvés, lesquels sont pairs ? impairs ?

    \item Écrire chacun des entiers naturels pairs sous forme $2k$ avec $k$ un entier naturel.
    \item Écrire chacun des entiers naturels impairs sous forme $2k+1$ avec $k$ un entier naturel.
\end{enumerate}
\end{Activite}

%% APPLICATION
\begin{Application}[][15 min]
Soit $n$ un entier naturel non nul.
\begin{enumerate}
    \item Étudier la parité de chacun des nombres suivants :
    \begin{multicols}{2}
    \begin{itemize}[label=\textbullet]
        \item $A=8n+10$
        \item $B=8n-1$
        \item $C=n(2n+4)$
        \item $D=4n^2+6n+7$
    \end{itemize}
    \end{multicols}
    \item
    \begin{enumerate}
        \item Montrer que $(n + 1)^2 - n^2$ est un nombre impair.
        \item Résoudre dans $\N$ l'équation : $(n + 1)^2 - n^2 = 2021$
    \end{enumerate}
\end{enumerate}
\end{Application}

%% APPLICATION
\begin{Application}[][15 min]
Soient $x$ et $y$ deux entiers naturels, tels que $x$ est pair et $y$ est impair. Déterminer la parité des nombres suivants :
\begin{center}
\begin{multicols}{2}
\begin{itemize}
    \item $x + 8y$
    \item $(x + 1)(y + 1)$
    \item $3x^3 + y^2 + 11$
    \item $x^{2024} + xy + 17$
\end{itemize}
\end{multicols}
\end{center}
\end{Application}

%% APPLICATION
\begin{Application}[][15 min]
Soit $n$ un entier naturel.
\begin{enumerate}
    \item Montrer que le nombre $n(n+1)$ est pair.
    \item Déterminer la parité des nombres suivants :
\begin{center}
$a=n^2+5n \quad ; \quad b=n^2+7n+13 \quad ; \quad c=n^4-n^2$
\end{center}
\item En déduire la valeur de $t$ :
\[  t=5\times(-1)^a-8\times(-1)^b+11\times(-1)^c \]
\end{enumerate}
\end{Application}

%% APPLICATION
\begin{Application}[][15 min]
\begin{enumerate}
    \item Déterminer les multiples de $13$ compris entre $100$ et $199$.
    \item Montrer que $3333$ est un multiple de $33$.
\end{enumerate}
\end{Application}

%% APPLICATION
\begin{Application}[][15 min]
\begin{enumerate}
    \item Déterminer parmi les nombres suivants ceux qui sont divisibles à la fois par $2$ et par $3$ :

    $126 \quad ; \quad 345 \quad ; \quad 810 \quad ; \quad 442 \quad ; \quad 3054 \quad ; \quad 5051 \quad ; \quad 7204$

    \item Déterminer parmi les nombres suivants ceux qui sont divisibles à la fois par $5$ et par $9$ :

    $315 \quad ; \quad 450 \quad ; \quad 5550 \quad ; \quad 333 \quad ; \quad 9045 \quad ; \quad 3800 \quad ; \quad 9003$
\end{enumerate}
\end{Application}

%% APPLICATION
\begin{Application}[][15 min]
\begin{enumerate}
    \item Déterminer toutes les valeurs possibles du chiffre $x$ pour que le nombre $N = \overline{4x3}$ soit divisible par $3$.
    \item Déterminer le chiffre $y$ pour que le nombre $M = \overline{5y}$ soit à la fois pair et divisible par $9$.
\end{enumerate}
\end{Application}

%% ACTIVITE
\begin{Activite}[][10 min]
\begin{enumerate}
    \item Déterminer les diviseurs de chacun des entiers naturels : $2 \ ; \ 3 \ ; \ 4 \ ; \ 5 \ ; \ 6 \ ; \ 7 \ \text{et} \ 11$.

    \item Parmi ces nombres, relever ceux qui admettent \textbf{exactement deux} diviseurs.

    (\textit{On appelle ces nombres particuliers des \textbf{nombres premiers}}).

    \item Les nombres $0$ et $1$ sont-ils des nombres premiers ? Pourquoi ?

    \item Un nombre pair peut-il être premier ? Justifier.

    \item Le nombre $1071$ est-il premier ? Pourquoi ?
\end{enumerate}
\end{Activite}

%% APPLICATION
\begin{Application}[][15 min]
\begin{enumerate}
    \item Soit $n$ un entier naturel tel que $n \ge 2$.
          Montrer que le nombre $A = n^2 + n$ n'est pas premier.

    \item Soit $p$ un nombre premier tel que $p > 2$. Justifier que le nombre $p + 7$ n'est pas premier.
\end{enumerate}
\end{Application}

%% APPLICATION
\begin{Application}[][15 min]
  En utilisant le théorème du test de primalité, déterminer si les nombres suivants sont premiers : $127$ et $133$.
\end{Application}

%% ACTIVITE
\begin{Activite}[][10 min]
\begin{enumerate}
   \item Vérifier que $1400 = 2^3 \times 5^2 \times 7 $.

   \textit{(Cette écriture unique s'appelle la \textbf{décomposition en produit de facteurs premiers})}.
    \item Donner la décomposition du $864$ en produit de facteurs premiers.
    \item Quel est le plus petit entier qui multiplié par $864$, donne un carré ?
\end{enumerate}
\end{Activite}

%% APPLICATION
\begin{Application}[][20 min]
\begin{enumerate}
    \item Établir la décomposition en produit de facteurs premiers des nombres suivants : $A = 120$ et $B = 252$.

    \item En déduire la décomposition en produit de facteurs premiers du nombre $C = A \times B$.

    \item Quel est le plus petit entier naturel $k$ pour lequel $A \times k$ soit un carré parfait ?

    \item Justifier que $A$ est un multiple de $15$ mais qu'il n'est pas divisible par $9$.
\end{enumerate}
\end{Application}

%% ACTIVITE
\begin{Activite}[][10 min]
Deux bus partent de la gare en même temps.
Le bus A y repasse toutes les $15$ minutes et le bus B toutes les $20$ minutes.

\begin{enumerate}
    \item Déterminer les durées (en minutes) écoulées lors des cinq premiers passages du bus A.
    \item Déterminer les durées (en minutes) écoulées lors des cinq premiers passages du bus B.
    \item Au bout de combien de minutes les deux bus se retrouveront-ils à la gare en même temps pour la première fois ?
\end{enumerate}
\end{Activite}

\end{Application}

Un professeur d'EPS forme des équipes identiques (même nombre de filles et de garçons) avec $12$ filles et $18$ garçons. Tous les élèves participent.

\begin{enumerate}
    \item Peut-il former $3$ équipes ? Si oui, quelle sera la composition d'une équipe ?
    \item Peut-il former $4$ équipes ? Pourquoi ?
    \item Quel est le plus grand nombre d'équipes qu'il peut former ?
\end{enumerate}
\end{Activite}

On considère les deux nombres : $a = 3960$ ~\text{et}~ $b = 2100$
