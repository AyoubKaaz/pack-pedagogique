\begin{Solution}[ACTIVITÉ][1]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item Les entiers positifs sont $0 ~; 14 ~; 25 ~; 36$ et $41$.

    \begin{itemize}[label=$\blacktriangleright$, leftmargin=*, nosep]
    \item \textbf{Ces nombres constituent l'ensemble des \textcolor{PrimaryColor}{entiers naturels}, noté $\mathbb{N}$.}
    \end{itemize}

    \item Les nombres pairs sont $0 ; 14$ et $36$.

    Les nombres impairs sont $25$ et $41$.

    \begin{itemize}[label=$\blacktriangleright$, leftmargin=*, nosep]
    \item \textbf{Tout entier naturel est soit \textcolor{PrimaryColor}{pair}, soit \textcolor{PrimaryColor}{impair}.}
    \end{itemize}

    \item On a donc :
    \begin{itemize}[leftmargin=*,itemsep=2pt]
        \item $0 = 2 \times 0$ (avec $k=0$)

        \item $14 = 2 \times 7$ (avec $k=7$)

        \item $36 = 2 \times 18$ (avec $k=18$)
    \end{itemize}

    \begin{itemize}[label=$\blacktriangleright$, leftmargin=*, nosep]
    \item \textbf{Un entier naturel $n$ est \textcolor{PrimaryColor}{pair} s'il existe un entier naturel $k$ tel que $n = 2k$.}
    \end{itemize}

    \item On a donc :
    \begin{itemize}[leftmargin=*,itemsep=2pt]
        \item $25 = 24 + 1 = 2 \times 12 + 1$ (avec $k=12$)
        \item $41 = 40 + 1 = 2 \times 20 + 1$ (avec $k=20$)
    \end{itemize}

    \begin{itemize}[label=$\blacktriangleright$, leftmargin=*, nosep]
    \item \textbf{Un entier naturel $n$ est \textcolor{PrimaryColor}{impair} s'il existe un entier naturel $k$ tel que $n = 2k+1$.}
\end{itemize}
\end{enumerate}
\end{Solution}

\begin{Solution}[APPLICATION][1]
Soit $n \in \mathbb{N}$.
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item
    \begin{itemize}[leftmargin=*,itemsep=2pt]
        \item \textbf{Parité de $A = 8n + 10$ :}

    On a : $A = 8n + 10 = 2(4n + 5)$

    Posons $k = 4n + 5 \in \N$

    Donc $A = 2k$ avec $k \in \N$.

    Par conséquent, $A$ est un nombre \textbf{pair}.

        \item \textbf{Parité de $B = 8n - 1$ :}

    On a : $B = 8n - 1 = 2(4n - 1) + 1$

    Posons $k = 4n - 1 \in \N$ (car $n \geq 1$)

    Donc $B = 2k + 1$ avec $k \in \N$.

    Par conséquent, $B$ est un nombre \textbf{impair}.

        \item \textbf{Parité de $C = n(2n + 4)$ :}

    On a : $C = n(2n + 4) = 2n(n + 2)$

    Posons $k = n(n + 2) \in \N$

    Donc $C = 2k$ avec $k \in \N$.

    Donc $C$ est un nombre \textbf{pair}.


        \item \textbf{Parité de $D = 4n^2 + 6n + 7$ :}

    On a :
    \begin{align*}
        D = 4n^2 + 6n + 7
          = 4n^2 + 6n + 6 + 1
          = 2(2n^2 + 3n + 3) + 1
    \end{align*}

    Posons $k = 2n^2 + 3n + 3 \in \N$

    Donc $D = 2k + 1$ avec $k \in \N$.

    Donc  $D$ est un nombre \textbf{impair}.
    \end{itemize}

    \item
    \begin{enumerate}[leftmargin=*,itemsep=2pt]
        \item \textbf{Montrons que $(n+1)^2 - n^2$ est impair :}

        Soit $n \in \N$,  on a :
        \begin{align*}
            (n+1)^2 - n^2 = n^2 + 2n + 1 - n^2
                          = 2n + 1
        \end{align*}

        Donc $(n+1)^2 - n^2$ est un nombre \textbf{impair}.

        \item \textbf{Résolution dans $\N$ de $(n+1)^2 - n^2 = 2021$ :}

        D'après la question \textbf{a)}, on a $(n+1)^2 - n^2 = 2n + 1$.

        Donc :
        \begin{align*}
            (n+1)^2 - n^2 = 2021
            \text{ équivaut à } \quad & 2n + 1 = 2021 \\
            \text{ équivaut à } \quad & 2n = 2021 - 1 \\
            \text{ équivaut à } \quad & 2n = 2020 \\
            \text{ équivaut à } \quad & n = \frac{2020}{2} \\
            \text{ équivaut à } \quad & n = 1010
        \end{align*}
        Puisque $1010 \in \mathbb{N}$, \\
        Donc : \quad S = \{1010\}

    \end{enumerate}
\end{enumerate}
\end{Solution}

\begin{Solution}[APPLICATION][2]
Soient $x \in \mathbb{N}$ et $y \in \mathbb{N}$. \\
Puisque $x$ est pair et $y$ est impair, on applique les propriétés des opérations sur les nombres pairs et impairs :
\begin{itemize}[leftmargin=*,itemsep=2pt]
    \item Puisque $8$ est un nombre pair, alors le produit $8y$ est pair. \\
    Alors la somme $x + 8y$ est paire. \\
    Donc : \quad $x + 8y \text{ est un nombre pair.}$

    \item Puisque $x$ est pair et $1$ est impair, alors $x + 1$ est impair. \\
    Puisque $y$ est impair et $1$ est impair, alors $y + 1$ est pair. \\
    Donc le produit $(x + 1)(y + 1)$ est pair.

    \item Puisque $x$ est pair, alors $x^3$ est pair. \\
    Donc le produit $3x^3$ est pair. \\
    Puisque $y$ est impair, alors $y^2$ est impair. \\
    Donc la somme $y^2 + 11$ est paire. \\
    D'où la somme $3x^3 + y^2 + 11$ est paire.

    \item Puisque $x$ est pair, alors $x^{2024}$ est pair et le produit $xy$ est pair. \\
    Donc la somme $x^{2024} + xy$ est paire. \\
    Puisque $x^{2024} + xy$ est pair et $17$ est impair, alors la somme totale est impaire. \\
    Donc : \quad $x^{2024} + xy + 17 \text{ est un nombre impair.}$
\end{itemize}
\end{Solution}

\begin{Solution}[APPLICATION][3]
Soit $n \in \mathbb{N}$.
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item \textbf{Montrons que $n(n+1)$ est pair :}

    Parmi deux entiers consécutifs $n$ et $n+1$, l'un est pair et l'autre est impair.

    Donc leur produit contient un facteur pair.

    Par conséquent, $n(n+1)$ est un nombre \textbf{pair}.

    \item \textbf{Parité de $a$, $b$ et $c$ :}
    \begin{itemize}[leftmargin=*,itemsep=2pt]
        \item \textbf{Pour $a = n^2 + 5n$ :}

        On a : $a = n^2 + n + 4n = n(n+1) + 2(2n)$

        Puisque $n(n+1)$ est pair et $2(2n)$ est pair, leur somme est paire.

        Donc $a$ \text{ est pair}.

        \item \textbf{Pour $b = n^2 + 7n + 13$ :}

        On a : $b = n^2 + n + 6n + 12 + 1 = n(n+1) + 2(3n + 6) + 1$

        Puisque $n(n+1)$ est pair et $2(3n+6)$ est pair, leur somme est paire.

        Donc  b \text{ est impair}.

        \item \textbf{Pour $c = n^4 - n^2$ :}

        On a :
        \begin{align*}
c = n^2(n^2 - 1)
 = n^2(n-1)(n+1)
 = n(n-1) \times n(n+1)
\end{align*}

        Puisque $n(n-1)$ est le produit de deux entiers consécutifs, il est pair.

        D'où : $c$ est le produit d'un nombre pair par un autre entier.

        Donc : \quad c \text{ est pair}.
    \end{itemize}

    \item \textbf{Déduisons la valeur de $t$ :}

    On a : $t = 5 \times (-1)^a - 8 \times (-1)^b + 11 \times (-1)^c$

    Puisque $a$ est pair, $(-1)^a = 1$

    Puisque $b$ est impair, $(-1)^b = -1$

    Puisque $c$ est pair, $(-1)^c = 1$

    On a donc :
    $t = 5 \times 1 - 8 \times (-1) + 11 \times 1
     = 5 + 8 + 11$

    D'où : \quad $t = 24$
\end{enumerate}
\end{Solution}

\begin{Solution}[APPLICATION][6]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item \textbf{Valeurs de $x$ pour que $N = \overline{4x3}$ soit divisible par $3$ :}

    $N$ est divisible par $3$ si et seulement si la somme de ses chiffres est divisible par $3$.

    On a : $4 + x + 3 = 7 + x$

    Il faut donc que $3$ divise $(7 + x)$, avec $x \in \{0 \; ; \; 1 \; ; \; \ldots \; ; \; 9\}$.

    Puisque $0 \leq x \leq 9$, alors : $7 \leq 7 + x \leq 16$.

    Les multiples de $3$ compris entre $7$ et $16$ sont : $9$, $12$ et $15$.

    On a donc :
    \begin{align*}
        7 + x &= 9 \quad \text{ équivaut à } \quad x = 2 \\
        7 + x &= 12 \quad \text{ équivaut à } \quad x = 5 \\
        7 + x &= 15 \quad \text{ équivaut à } \quad x = 8
    \end{align*}

    Donc : \quad les valeurs possibles du chiffre $x$ sont $2$, $5$ et $8$.

    \item \textbf{Valeur de $y$ pour que $M = \overline{5y}$ soit pair et divisible par $9$ :}

    Puisque $M$ est pair, son chiffre des unités $y$ est pair.

    On a donc : $y \in \{0, 2, 4, 6, 8\}$.

    Puisque $M$ est divisible par $9$, la somme de ses chiffres est un multiple de $9$.

    On a donc : $5 + y$ est un multiple de $9$.

    Puisque $0 \leq y \leq 8$, alors : $5 \leq 5 + y \leq 13$.

    Le seul multiple de $9$ compris entre $5$ et $13$ est $9$.

    On a donc :
    \begin{align*}
        5 + y &= 9 \quad \text{ équivaut à } \quad y = 4
    \end{align*}

    Puisque $4$ est un chiffre pair, cette valeur convient.

    Par conséquent, $y = 4$.
\end{enumerate}
\end{Solution}

\begin{Solution}[APPLICATION][7]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item \textbf{Montrons que $A = n^2 + n$ n'est pas premier (pour $n \geq 2$) :}

    On a : $A = n^2 + n = n(n + 1)$

    Or $n \geq 2$, donc $n \geq 2$ et $n + 1 \geq 3$.

    Ainsi, $A = n(n+1)$ est un produit de deux facteurs dont chacun est différent de $1$ et de $A$.

    Donc $A$ admet au moins quatre diviseurs distincts : $1$, $n$, $n+1$ et $A$.

    Par conséquent, $A$ n'est \textbf{pas un nombre premier}.

    \item \textbf{Justifions que $p + 7$ n'est pas premier :}

    On sait que $p$ est premier et $p > 2$, donc $p$ est impair (car $2$ est le seul nombre premier pair).

    Or $7$ est impair. Donc $p + 7$ est la somme de deux nombres impairs.

    Donc $p + 7$ est pair.

    De plus, $p \geq 3$, donc $p + 7 \geq 10 > 2$.

    Ainsi, le nombre $p + 7$ est un entier naturel strictement supérieur à $2$ qui est divisible par $2$.


    Par conséquent, $p + 7$ n'est \textbf{pas un nombre premier}.
\end{enumerate}
\end{Solution}
