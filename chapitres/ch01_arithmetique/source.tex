%% SECTION
\section{Ensemble $\N$ --- Nombres pairs et nombres impairs \duree{15 min}}

%% ACTIVITE
\begin{Activite}
On considère la liste de nombres suivante : 
$-3 \ ; \ 14 \ ; \ 2,5 \ ; \ 0 \ ; \ 25 \ ; \ 36 \ ; \ -7 \ ; \ 41$

\begin{enumerate}
    \item Parmi les nombres de cette liste, relever ceux qui sont des entiers positifs.

    (\textit{Ces nombres sont appelés des \textbf{entiers naturels}}).
    
    \item Parmi les entiers naturels trouvés, lesquels sont pairs ? impairs ?  
        
    \item Écrire chacun des entiers naturels pairs sous forme $2k$ avec $k$ un entier naturel.
    \item Écrire chacun des entiers naturels impairs sous forme $2k+1$ avec $k$ un entier naturel.
\end{enumerate}
\end{Activite}

%% SOLUTION_ACTIVITE
\begin{Solution}[ACTIVITÉ][1]
\begin{enumerate}
    \item Les entiers positifs sont $0 ~; 14 ~; 25 ~; 36$ et $41$.

    $\blacktriangleright$ \textbf{Ces nombres constituent l'ensemble des \textcolor{PrimaryColor}{entiers naturels}, noté $\mathbb{N}$.}
    
    \item Les nombres pairs sont $0 ; 14$ et $36$ et les nombres impairs sont $25$ et $41$.

    $\blacktriangleright$ \textbf{Tout entier naturel est soit \textcolor{PrimaryColor}{pair}, soit \textcolor{PrimaryColor}{impair}.}
    
    \item On a donc :
    $0 = 2 \times 0$ (avec $k=0$) \quad ; \quad $14 = 2 \times 7$ (avec $k=7$) \quad et \quad $36 = 2 \times 18$ (avec $k=18$)
    
    
    $\blacktriangleright$ \textbf{Un entier naturel $n$ est \textcolor{PrimaryColor}{pair} s'il existe un entier naturel $k$ tel que $n = 2k$.}
    \item On a donc :
    \begin{itemize}
        \item $25 = 24 + 1 = 2 \times 12 + 1$ (avec $k=12$)
        \item $41 = 40 + 1 = 2 \times 20 + 1$ (avec $k=20$)
    \end{itemize}
    
    $\blacktriangleright$ \textbf{Un entier naturel $n$ est \textcolor{PrimaryColor}{impair} s'il existe un entier naturel $k$ tel que $n = 2k+1$.}
\end{enumerate}
\end{Solution}

%% DEFINITION
\begin{Definition}[]
\begin{itemize}
    \item On désigne par $\N$ l'\textbf{ensemble des entiers naturels}, et on écrit :
$\N = \{0\,;\,1\,;\,2\,;\,3\,;\,4\,;\ldots\}$
    
    \item On désigne par $\N^*$ l'\textbf{ensemble des entiers naturels non nuls}, et on écrit :
$\N^* = \{\,1\,;\,2\,;\,3\,;\,4\,;\ldots\}$
\end{itemize}
\end{Definition}



%% DEFINITION
\begin{Definition}[Nombres pairs et nombres impairs]
\begin{itemize}
  \item Un \textbf{nombre pair} est un multiple de $2$.
  \item Un \textbf{nombre impair} est un nombre qui n'est pas pair.
\end{itemize}
\end{Definition}



%% REMARQUE
\begin{Remarque}
   Il existe une infinité d'entiers naturels.
\end{Remarque}

%% PROPRIETE
\begin{Propriete}[Écriture d'un nombre pair ou impair quelconque] Soit $n$ un entier naturel.
\begin{itemize}
    \item $n$ est \textbf{pair} si et seulement s'il s'écrit :
$n = 2k \quad \text{avec } k \in \N$

    \item $n$ est \textbf{impair} si et seulement s'il s'écrit :
$n = 2k + 1 \quad \text{avec } k \in \N$
\end{itemize}
\end{Propriete}

%% EXEMPLE
\begin{Exemple}
\begin{enumerate}
    \item $2022$ est un nombre pair. Car $2022 = 2 \times 1011$
    \item $2023$ est un nombre impair. Car $2023 = 2 \times 1011 + 1$
    \item Soit $n \in \N^*$, 
    On pose : $E = 4n + 11$ et $F = 2n + 6$
\begin{multicols}{2}
    \begin{itemize}
        \item
On a $E = 4n + 11 = 4n + 10 + 1 = 2(2n + 5) + 1$

On pose $K = 2n + 5 \in \N$

D'où $E = 2K + 1$ avec $K \in \N$

Donc $E$ est un nombre impair.

        \columnbreak
        
        \item On a $F = 2n + 6 = 2(n + 3)$
    
    On pose $K = n + 3 \in \N$
    
    Donc $F = 2K$ avec $K \in \N$
    
    D'où $F$ est un nombre pair.
    \end{itemize}
\end{multicols}
\end{enumerate}
\end{Exemple}


%% APPLICATION
\begin{Application}[]
Soit $n$ un entier naturel. 
\begin{enumerate}
    \item Étudier la parité de chacun des nombres suivants :
    \begin{multicols}{4}
    \begin{itemize}[label=\textbullet]
        \item $A=8n+10$
        \item $B=8n-1$
        \item $C=n(2n+4)$
        \item $D=4n^2+6n+7$
    \end{itemize}
    \end{multicols}
    \item 
    \begin{enumerate}
        \item Montrer que $(n + 1)^2 - n^2$ est un nombre impair.
        \item Résoudre dans $\N$ l'équation : $(n + 1)^2 - n^2 = 2021$
    \end{enumerate}
\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][1]
\begin{enumerate}
    \item \textbf{Parité de $A = 8n + 10$ :}

    On a : $A = 8n + 10 = 2(4n + 5)$

    Posons $k = 4n + 5 \in \N$

    Donc $A = 2k$ avec $k \in \N$.

    Par conséquent, $A$ est un nombre \textbf{pair}.

    \medskip

    \textbf{Parité de $B = 8n - 1$ :}

    On a : $B = 8n - 1 = 2(4n - 1) + 1$

    Posons $k = 4n - 1 \in \N$ (pour $n \geq 1$)

    Donc $B = 2k + 1$ avec $k \in \N$.

    Par conséquent, $B$ est un nombre \textbf{impair}.

    \medskip

    \textbf{Parité de $C = n(2n + 4)$ :}

    On a : $C = n(2n + 4) = 2n(n + 2)$

    Posons $k = n(n + 2) \in \N$

    Donc $C = 2k$ avec $k \in \N$.

    Par conséquent, $C$ est un nombre \textbf{pair}.

    \medskip

    \textbf{Parité de $D = 4n^2 + 6n + 7$ :}

    On a :
    \begin{align*}
        D &= 4n^2 + 6n + 7 \\
          &= 4n^2 + 6n + 6 + 1 \\
          &= 2(2n^2 + 3n + 3) + 1
    \end{align*}

    Posons $k = 2n^2 + 3n + 3 \in \N$

    Donc $D = 2k + 1$ avec $k \in \N$.

    Par conséquent, $D$ est un nombre \textbf{impair}.

    \item
    \begin{enumerate}
        \item \textbf{Montrons que $(n+1)^2 - n^2$ est impair :}

        On a :
        \begin{align*}
            (n+1)^2 - n^2 &= n^2 + 2n + 1 - n^2 \\
                          &= 2n + 1
        \end{align*}

        Posons $k = n \in \N$

        Donc $(n+1)^2 - n^2 = 2k + 1$ avec $k \in \N$.

        Par conséquent, $(n+1)^2 - n^2$ est un nombre \textbf{impair}.

        \item \textbf{Résolution dans $\N$ de $(n+1)^2 - n^2 = 2021$ :}

        D'après la question \textbf{a)}, on a $(n+1)^2 - n^2 = 2n + 1$.

        On a :
        \[2n + 1 = 2021\]
        \[\Leftrightarrow 2n = 2020 \qquad \text{(on soustrait 1 aux deux membres)}\]
        \[\Leftrightarrow n = 1010 \qquad \text{(on divise par 2,}\ 2 > 0\text{)}\]

        Or $1010 \in \N$.

        Par conséquent, $S = \{1010\}$.
    \end{enumerate}
\end{enumerate}
\end{Solution}

%% PROPRIETE
\begin{Propriete}[Opérations sur les nombres pairs et impairs]
\begin{itemize}
  \item La somme de deux nombres de \textbf{même parité} est \textbf{paire}.
  \item La somme de deux nombres de \textbf{parité différente} est \textbf{impaire}.
  \item Si l'un des facteurs d'un produit est pair, alors ce produit est \textbf{pair}.
  \item Tout entier naturel élevé à une puissance non nulle conserve sa parité.
\end{itemize}
\end{Propriete}

%% APPLICATION
\begin{Application}[]
Soient $x$  et $y$ deux entiers naturels, tels que $x$ est pair et $y$ est impair. Déterminer la parité des nombres suivants :
\begin{center}
\begin{multicols}{4}
\begin{itemize}
    \item $x + 8y$
    \item $(x + 1)(y + 1)$
    \item $3x^3 + y^2 + 11$
    \item $x^{2024} + xy + 17$
\end{itemize}
\end{multicols}
\end{center}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][2]

On pose : $x = 2a$ avec $a \in \N$ (car $x$ est pair) et $y = 2b + 1$ avec $b \in \N$ (car $y$ est impair).

\textbf{Parité de $x + 8y$ :}

On a :
\begin{align*}
    x + 8y &= 2a + 8(2b + 1) \\
           &= 2a + 16b + 8 \\
           &= 2(a + 8b + 4)
\end{align*}

Posons $k = a + 8b + 4 \in \N$

Donc $x + 8y = 2k$ avec $k \in \N$.

Par conséquent, $x + 8y$ est un nombre \textbf{pair}.

\medskip

\textbf{Parité de $(x+1)(y+1)$ :}

On a :
\begin{align*}
    (x+1)(y+1) &= (2a + 1)(2b + 2) \\
               &= (2a + 1) \cdot 2(b + 1) \\
               &= 2(2a + 1)(b + 1)
\end{align*}

Posons $k = (2a + 1)(b + 1) \in \N$

Donc $(x+1)(y+1) = 2k$ avec $k \in \N$.

Par conséquent, $(x+1)(y+1)$ est un nombre \textbf{pair}.

\medskip

\textbf{Parité de $3x^3 + y^2 + 11$ :}

On a :
\begin{align*}
    3x^3 + y^2 + 11 &= 3(2a)^3 + (2b+1)^2 + 11 \\
                    &= 24a^3 + 4b^2 + 4b + 1 + 11 \\
                    &= 24a^3 + 4b^2 + 4b + 12 \\
                    &= 2(12a^3 + 2b^2 + 2b + 6)
\end{align*}

Posons $k = 12a^3 + 2b^2 + 2b + 6 \in \N$

Donc $3x^3 + y^2 + 11 = 2k$ avec $k \in \N$.

Par conséquent, $3x^3 + y^2 + 11$ est un nombre \textbf{pair}.

\medskip

\textbf{Parité de $x^{2024} + xy + 17$ :}

On a :
\begin{align*}
    x^{2024} + xy + 17 &= (2a)^{2024} + (2a)(2b+1) + 17 \\
                       &= 2^{2024}a^{2024} + 4ab + 2a + 17 \\
                       &= 2^{2024}a^{2024} + 4ab + 2a + 16 + 1 \\
                       &= 2(2^{2023}a^{2024} + 2ab + a + 8) + 1
\end{align*}

Posons $k = 2^{2023}a^{2024} + 2ab + a + 8 \in \N$

Donc $x^{2024} + xy + 17 = 2k + 1$ avec $k \in \N$.

Par conséquent, $x^{2024} + xy + 17$ est un nombre \textbf{impair}.
\end{Solution}

%% PROPRIETE
\begin{Propriete}[Nombres consécutifs]
Soient $n$ et $n+1$ deux entiers consécutifs.
\begin{itemize}
  \item La somme de deux entiers consécutifs est \textbf{impaire}.
  \item Le produit de deux entiers consécutifs est \textbf{pair}.
\end{itemize}
\end{Propriete}


%% APPLICATION
\begin{Application}[]
Soit $n$ un entier naturel.
\begin{enumerate}
    \item Montrer que le nombre $n(n+1)$ est pair.
    \item Déterminer la parité des nombres suivants :
\begin{center}
$a=n^2+5n \quad ; \quad b=n^2+7n+13 \quad ; \quad c=n^4-n^2$
\end{center}
\item En déduire la valeur de $t$ :
\[  t=5\times(-1)^a-8\times(-1)^b+11\times(-1)^c \]
\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][3]
\begin{enumerate}
    \item \textbf{Montrons que $n(n+1)$ est pair :}

    Parmi deux entiers consécutifs $n$ et $n+1$, l'un est pair et l'autre est impair.

    Donc leur produit contient un facteur pair.

    Il existe donc un entier $A \in \N$ tel que $n(n+1) = 2A$.

    Par conséquent, $n(n+1)$ est un nombre \textbf{pair}.

    \item \textbf{Parité de $a$, $b$ et $c$ :}

    \textbf{Parité de $a = n^2 + 5n$ :}

    On a : $a = n^2 + 5n = n^2 + n + 4n = n(n+1) + 4n$

    D'après la question \textbf{1)}, $n(n+1) = 2A$.

    Donc : $a = 2A + 4n = 2(A + 2n)$

    Posons $k = A + 2n \in \N$

    Donc $a = 2k$ avec $k \in \N$.

    Par conséquent, $a$ est un nombre \textbf{pair}.

    \medskip

    \textbf{Parité de $b = n^2 + 7n + 13$ :}

    On a : $b = n^2 + 7n + 13 = n^2 + n + 6n + 12 + 1 = n(n+1) + 6n + 12 + 1$

    D'après la question \textbf{1)}, $n(n+1) = 2A$.

    Donc : $b = 2A + 6n + 12 + 1 = 2(A + 3n + 6) + 1$

    Posons $k = A + 3n + 6 \in \N$

    Donc $b = 2k + 1$ avec $k \in \N$.

    Par conséquent, $b$ est un nombre \textbf{impair}.

    \medskip

    \textbf{Parité de $c = n^4 - n^2$ :}

    On a : $c = n^4 - n^2 = n^2(n^2 - 1) = n^2(n-1)(n+1)$

    D'après la question \textbf{1)}, $n(n+1) = 2A$, donc $n^2(n+1) = 2An$.

    Ainsi : $c = (n-1) \times n^2(n+1) = (n-1) \times 2An = 2An(n-1)$

    Posons $k = An(n-1) \in \N$

    Donc $c = 2k$ avec $k \in \N$.

    Par conséquent, $c$ est un nombre \textbf{pair}.

    \item \textbf{Valeur de $t$ :}

    On a : $a$ est pair $\Rightarrow (-1)^a = 1$ \quad ; \quad $b$ est impair $\Rightarrow (-1)^b = -1$ \quad ; \quad $c$ est pair $\Rightarrow (-1)^c = 1$.

    Donc :
    \begin{align*}
        t &= 5 \times (-1)^a - 8 \times (-1)^b + 11 \times (-1)^c \\
          &= 5 \times 1 - 8 \times (-1) + 11 \times 1 \\
          &= 5 + 8 + 11
    \end{align*}

    Par conséquent, $t = 24$.
\end{enumerate}
\end{Solution}

\section{Multiples et diviseurs d'un entier naturel \duree{15 min}}

%% DEFINITION
\begin{Definition}[Multiples et diviseurs]
Soient $a$ et $b$ deux entiers naturels.

S'il existe un entier naturel $k$ tel que $a = k \times b$, on dit que :
\begin{itemize}
  \item $b$ \textbf{divise} $a$ (ou $b$ est un \textbf{diviseur} de $a$),
  \item $a$ est \textbf{divisible} par $b$,
  \item $a$ est un \textbf{multiple} de $b$.
\end{itemize}
\end{Definition}

%% REMARQUE
\begin{Remarque}
\begin{itemize}
  \item $0$ est un multiple de tous les nombres entiers naturels.
  \item Tout entier naturel est divisible par $1$ et par lui-même.
\end{itemize}
\end{Remarque}

%% EXEMPLE
\begin{Exemple}
\begin{enumerate}
    \item On a $145 = 5 \times 29$ ; donc $145$ est un multiple de $5$ et de $29$.
    
    \item Le nombre $40$ n'est pas divisible par $7$.
    
    \item Les multiples de $30$ sont : $0 ; 30 ; 60 ; 90 ; \dots$ 
    (il y en a une infinité).
    
    \item Les diviseurs de $30$ sont :
    $ D_{30} = \{1 ; 2 ; 3 ; 5 ; 6 ; 10 ; 15 ; 30\} $
    (il y en a un nombre fini).
\end{enumerate}
\end{Exemple}


%% APPLICATION
\begin{Application}[]
\begin{enumerate}
    \item Déterminer les multiples de $13$ compris entre $100$ et $199$.
    \item Montrer que $3333$ est un multiple de $33$.
\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][4]
\begin{enumerate}
    \item \textbf{Multiples de $13$ compris entre $100$ et $199$ :}

    On cherche les entiers $k \in \N$ tels que $100 \leq 13k \leq 199$.

    On a : $\frac{100}{13} \approx 7{,}69$ et $\frac{199}{13} \approx 15{,}30$.

    Donc $k \in \{8 \; ; \; 9 \; ; \; 10 \; ; \; 11 \; ; \; 12 \; ; \; 13 \; ; \; 14 \; ; \; 15\}$.

    Les multiples de $13$ compris entre $100$ et $199$ sont :
    \[104 \; ; \; 117 \; ; \; 130 \; ; \; 143 \; ; \; 156 \; ; \; 169 \; ; \; 182 \; ; \; 195\]

    \item \textbf{Montrons que $3333$ est un multiple de $33$ :}

    On a : $3333 = 33 \times 101$

    Posons $K = 101 \in \N$

    Donc $3333 = 33 \times K$ avec $K \in \N$.

    Par conséquent, $3333$ est un \textbf{multiple de $33$}.
\end{enumerate}
\end{Solution}

%% PROPRIETE
\begin{Propriete}[Opérations sur les multiples]
Soient $a$, $b$ et $c$ des entiers naturels.
\begin{itemize}
  \item Si $a$ est multiple de $c$ et $b$ est multiple de $c$, alors $a + b$ est multiple de $c$ ;
        
        et si de plus $a \geq b$, alors $a - b$ est multiple de $c$.
  \item Si $a$ est multiple de $b$ et $b$ est multiple de $c$, alors $a$ est multiple de $c$.
  \item Si $c$ est multiple de $a$, alors $nc$ est multiple de $a$ pour tout $n \in \N$.
\end{itemize}
\end{Propriete}

%% PROPRIETE
\begin{Propriete}[Opérations sur les diviseurs]
Soient $a$, $b$ et $c$ des entiers naturels.
\begin{itemize}
  \item Si $a$ divise $b$ et $c$, et $b \geq c$, alors $a$ divise $b + c$ et $b - c$.
  \item Si $a$ divise $b$, alors $a$ divise $bc$.
\end{itemize}
\end{Propriete}

%% EXEMPLE
\begin{Exemple}
\begin{itemize}
    \item $20$ et $15$ sont des multiples de $5$, donc $20 + 15 = 35$ et $20 - 15 = 5$ sont aussi des multiples de $5$.
    \item $24$ est un multiple de $12$ et $12$ est un multiple de $3$, donc $24$ est un multiple de $3$.
    \item $8$ est un multiple de $4$, donc $8 \times 5 = 40$ est un multiple de $4$.
    \item $3$ divise $15$ et $9$, donc $3$ divise $15 + 9 = 24$ et $15 - 9 = 6$.
    \item $7$ divise $14$, donc $7$ divise $14 \times 2 = 28$.
\end{itemize}
\end{Exemple}

%% PROPRIETE
\begin{Propriete}[Critères de divisibilité] Un nombre est divisible par :
\begin{itemize}
  \item $2$ si son chiffre des unités est pair.
  \item $3$ si la somme de ses chiffres est divisible par $3$.
  \item $5$ si son chiffre des unités est $0$ ou $5$.
  \item $9$ si la somme de ses chiffres est divisible par $9$.
\end{itemize}
\end{Propriete}

%% APPLICATION
\begin{Application}[]
\begin{enumerate}
    \item Déterminer parmi les nombres suivants ceux qui sont divisibles à la fois par $2$ et par $3$ :
\begin{center}
    $126 \quad ; \quad 345 \quad ; \quad 810 \quad ; \quad 442 \quad ; \quad 3054 \quad ; \quad 5051 \quad ; \quad 7204$
\end{center}
    \item Déterminer parmi les nombres suivants ceux qui sont divisibles à la fois par $5$ et par $9$ :
\begin{center}
    $315 \quad ; \quad 450 \quad ; \quad 5550 \quad ; \quad 333 \quad ; \quad 9045 \quad ; \quad 3800 \quad ; \quad 9003$
\end{center}
\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][5]
\begin{enumerate}
    \item \textbf{Divisibles à la fois par $2$ et par $3$ :}

    \begin{itemize}
        \item $126$ : unités $= 6$ (pair) ; $1+2+6=9$ (div. par $3$) $\Rightarrow$ \textbf{oui}
        \item $345$ : unités $= 5$ (impair) $\Rightarrow$ \textbf{non}
        \item $810$ : unités $= 0$ (pair) ; $8+1+0=9$ (div. par $3$) $\Rightarrow$ \textbf{oui}
        \item $442$ : unités $= 2$ (pair) ; $4+4+2=10$ (non div. par $3$) $\Rightarrow$ \textbf{non}
        \item $3054$ : unités $= 4$ (pair) ; $3+0+5+4=12$ (div. par $3$) $\Rightarrow$ \textbf{oui}
        \item $5051$ : unités $= 1$ (impair) $\Rightarrow$ \textbf{non}
        \item $7204$ : unités $= 4$ (pair) ; $7+2+0+4=13$ (non div. par $3$) $\Rightarrow$ \textbf{non}
    \end{itemize}

    Par conséquent, les nombres divisibles à la fois par $2$ et par $3$ sont : $\mathbf{126}$, $\mathbf{810}$ et $\mathbf{3054}$.

    \item \textbf{Divisibles à la fois par $5$ et par $9$ :}

    \begin{itemize}
        \item $315$ : unités $= 5$ (div. par $5$) ; $3+1+5=9$ (div. par $9$) $\Rightarrow$ \textbf{oui}
        \item $450$ : unités $= 0$ (div. par $5$) ; $4+5+0=9$ (div. par $9$) $\Rightarrow$ \textbf{oui}
        \item $5550$ : unités $= 0$ (div. par $5$) ; $5+5+5+0=15$ (non div. par $9$) $\Rightarrow$ \textbf{non}
        \item $333$ : unités $= 3$ (non div. par $5$) $\Rightarrow$ \textbf{non}
        \item $9045$ : unités $= 5$ (div. par $5$) ; $9+0+4+5=18$ (div. par $9$) $\Rightarrow$ \textbf{oui}
        \item $3800$ : unités $= 0$ (div. par $5$) ; $3+8+0+0=11$ (non div. par $9$) $\Rightarrow$ \textbf{non}
        \item $9003$ : unités $= 3$ (non div. par $5$) $\Rightarrow$ \textbf{non}
    \end{itemize}

    Par conséquent, les nombres divisibles à la fois par $5$ et par $9$ sont : $\mathbf{315}$, $\mathbf{450}$ et $\mathbf{9045}$.
\end{enumerate}
\end{Solution}

%% APPLICATION
\begin{Application}
\begin{enumerate}
    \item Déterminer toutes les valeurs possibles du chiffre $x$ pour que le nombre $N = \overline{4x3}$ soit divisible par $3$.
    \item Déterminer le chiffre $y$ pour que le nombre $M = \overline{5y}$ soit à la fois pair et divisible par $9$.
\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][6]
\begin{enumerate}
    \item \textbf{Valeurs de $x$ pour que $N = \overline{4x3}$ soit divisible par $3$ :}

    D'après le critère de divisibilité par $3$, $N$ est divisible par $3$ si et seulement si la somme de ses chiffres est divisible par $3$.

    On a : somme des chiffres $= 4 + x + 3 = 7 + x$.

    Il faut donc que $3 \mid (7 + x)$, avec $x \in \{0 \; ; \; 1 \; ; \; \ldots \; ; \; 9\}$.

    \begin{itemize}
        \item $x = 2$ : $7 + 2 = 9 = 3 \times 3$ \quad \textbf{oui}
        \item $x = 5$ : $7 + 5 = 12 = 3 \times 4$ \quad \textbf{oui}
        \item $x = 8$ : $7 + 8 = 15 = 3 \times 5$ \quad \textbf{oui}
    \end{itemize}

    Par conséquent, les valeurs possibles de $x$ sont : $x \in \{2 \; ; \; 5 \; ; \; 8\}$.

    \item \textbf{Valeur de $y$ pour que $M = \overline{5y}$ soit pair et divisible par $9$ :}

    $M$ est pair $\Leftrightarrow$ le chiffre des unités $y$ est pair, donc $y \in \{0 \; ; \; 2 \; ; \; 4 \; ; \; 6 \; ; \; 8\}$.

    $M$ est divisible par $9$ $\Leftrightarrow$ $5 + y$ est divisible par $9$.

    On cherche $y$ pair tel que $5 + y \in \{9 \; ; \; 18\}$ :

    \begin{itemize}
        \item $5 + y = 9 \Rightarrow y = 4$ \quad (pair) \quad \textbf{oui}
        \item $5 + y = 18 \Rightarrow y = 13$ \quad (impossible, $y$ est un chiffre)
    \end{itemize}

    Par conséquent, $y = 4$.
\end{enumerate}
\end{Solution}

\section{Les nombres premiers \duree{15 min}}

%% ACTIVITE
\begin{Activite}
\begin{enumerate}
    \item Déterminer les diviseurs de chacun des entiers naturels : $2 \ ; \ 3 \ ; \ 4 \ ; \ 5 \ ; \ 6 \ ; \ 7 \ \text{et} \ 11$.
    
    \item Parmi ces nombres, relever ceux qui admettent \textbf{exactement deux} diviseurs.
    
    (\textit{On appelle ces nombres particuliers des \textbf{nombres premiers}}).
    
    \item Les nombres $0$ et $1$ sont-ils des nombres premiers ? Pourquoi ?
    
    \item Un nombre pair peut-il être premier ? Justifier.
    
    \item Le nombre $1071$ est-il premier ? Pourquoi ?
\end{enumerate}
\end{Activite}


%% SOLUTION_ACTIVITE
\begin{Solution}[ACTIVITÉ][2]

\begin{enumerate}
    \item Les diviseurs de chaque entier naturel :
\begin{multicols}{2}
\begin{itemize}
    \item $D_2 = \{1; 2\}$
    \item $D_3 = \{1; 3\}$
    \item $D_4 = \{1; 2; 4\}$
    \item $D_5 = \{1; 5\}$
    \item $D_6 = \{1; 2; 3; 6\}$
    \item $D_7 = \{1; 7\}$
    \item $D_{11} = \{1; 11\}$
\end{itemize}
\end{multicols}

    \item Les nombres admettant exactement deux diviseurs sont : $2$, $3$, $5$, $7$ et $11$.
    
    \item \begin{itemize}
    \item $0$ n'est pas premier : tout entier naturel non nul divise $0$, donc $0$ a une infinité de diviseurs.
    \item $1$ n'est pas premier : il n'admet qu'un seul diviseur (lui-même).
\end{itemize}
    
    \item Le nombre $2$ est premier et pair. Tout autre nombre pair est divisible par $2$, donc admet au moins trois diviseurs ($1$, $2$ et lui-même). Par conséquent, $2$ est le seul nombre premier pair.
    
    \item On a : $1 + 0 + 7 + 1 = 9$, qui est divisible par $9$. Donc $9$ divise $1071$.

Par conséquent, $1071$ n'est pas un nombre premier.
\end{enumerate}

\end{Solution}

%% DEFINITION
\begin{Definition}[]
Un nombre est \textbf{premier} s'il possède \underline{exactement} deux diviseurs : $1$ et lui-même.
\end{Definition}

\begin{Application}[]

          Montrer que le nombre $A = n^2 + n$ n'est pas premier.
    
    \item Soit $p$ un nombre premier tel que $p > 2$. Justifier que le nombre $p + 7$ n'est pas premier.
\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][7]
\begin{enumerate}
    \item \textbf{Montrons que $A = n^2 + n$ n'est pas premier (pour $n \geq 2$) :}

    On a : $A = n^2 + n = n(n + 1)$

    Or $n \geq 2$, donc $n \geq 2$ et $n + 1 \geq 3$.

    Ainsi, $A = n(n+1)$ est un produit de deux facteurs dont chacun est différent de $1$ et de $A$.

    Donc $A$ admet au moins quatre diviseurs distincts : $1$, $n$, $n+1$ et $A$.

    Par conséquent, $A$ n'est \textbf{pas un nombre premier}.

    \item \textbf{Justifions que $p + 7$ n'est pas premier :}

    On sait que $p$ est premier et $p > 2$, donc $p$ est impair (car $2$ est le seul nombre premier pair).

    Or $7$ est impair. Donc $p + 7$ est la somme de deux nombres impairs.

    D'après la propriété : la somme de deux nombres impairs est paire.

    Donc $p + 7$ est pair.

    De plus, $p \geq 3$, donc $p + 7 \geq 10 > 2$.

    Ainsi, $p + 7$ est un nombre pair supérieur à $2$, donc divisible par $2$ sans être égal à $2$.

    Par conséquent, $p + 7$ n'est \textbf{pas un nombre premier}.
\end{enumerate}
\end{Solution}


%% PROPRIETE
\begin{Theoreme}[Test de primalité] Soit $n$ un entier naturel tel que $n \geq 2$.

Pour que $n$ soit premier, il faut et il suffit qu'aucun nombre premier $p$ vérifiant $p \leq \sqrt{n}$ ne divise $n$.
\end{Theoreme}

%% EXEMPLE
\begin{Exemple}
$\bullet~$\textbf{Montrons que le nombre $109$ est premier :}

On a : $\sqrt{109} \approx 10,4$ ;
les nombres premiers inférieurs ou égaux à $\sqrt{109}$ sont : $2$, $3$, $5$ et $7$.

Aucun de ces nombres ne divise $109$. Donc $109$ est premier.
\end{Exemple}


%% APPLICATION
\begin{Application}
  En utilisant le théorème du test de primalité, déterminer si les nombres suivants sont premiers : $127$ et $133$.

\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][8]

\textbf{Test de primalité de $127$ :}

On a $\sqrt{127} \approx 11{,}27$. Les nombres premiers $p \leq 11$ à tester sont : $2$, $3$, $5$, $7$ et $11$.

\begin{itemize}
    \item $127$ est impair, donc $2 \nmid 127$.
    \item $1 + 2 + 7 = 10$, non divisible par $3$, donc $3 \nmid 127$.
    \item Le chiffre des unités est $7$, donc $5 \nmid 127$.
    \item $127 = 18 \times 7 + 1$, donc $7 \nmid 127$.
    \item $127 = 11 \times 11 + 6$, donc $11 \nmid 127$.
\end{itemize}

Aucun nombre premier $p \leq \sqrt{127}$ ne divise $127$.

Par conséquent, $127$ est un nombre \textbf{premier}.

\medskip

\textbf{Test de primalité de $133$ :}

On a $\sqrt{133} \approx 11{,}53$. Les nombres premiers $p \leq 11$ à tester sont : $2$, $3$, $5$, $7$ et $11$.

\begin{itemize}
    \item $133$ est impair, donc $2 \nmid 133$.
    \item $1 + 3 + 3 = 7$, non divisible par $3$, donc $3 \nmid 133$.
    \item Le chiffre des unités est $3$, donc $5 \nmid 133$.
    \item $133 = 19 \times 7 + 0$, donc $7 \mid 133$.
\end{itemize}

On a : $133 = 7 \times 19$. Donc $133$ admet $7$ comme diviseur (autre que $1$ et $133$).

Par conséquent, $133$ n'est \textbf{pas un nombre premier}.

\end{Solution}

\section{Le plus petit commun multiple \duree{15 min}}


%% ACTIVITE
\begin{Activite}
Deux bus partent de la gare en même temps.
Le bus A y repasse toutes les $15$ minutes et le bus B toutes les $20$ minutes.

\begin{enumerate}
    \item Déterminer les durées (en minutes) écoulées lors des cinq premiers passages du bus A.

\end{enumerate}
\end{Activite}

%% SOLUTION_ACTIVITE
\begin{Solution}[ACTIVITÉ][4]
\begin{enumerate}
    \item Les durées écoulées en minutes lors des cinq premiers passages du bus A :
\[15 \; ; \; 30 \; ; \; 45 \; ; \; 60 \; ; \; 75 \quad (\text{sont les multiples de~} 15) \]

    \item Les durées écoulées en minutes lors des cinq premiers passages du bus B :
\[20 \; ; \; 40 \; ; \; 60 \; ; \; 80 \; ; \; 100 \quad (\text{sont les multiples de~} 20) \]

    \item Donc la plus petite durée commune aux deux bus est de 60 minutes.
    
    Alors : les deux bus se retrouveront à la gare en même temps pour la première fois au bout de 60 minutes.
    
    $\blacktriangleright$ \textbf{On dit que $60$ est le \textcolor{PrimaryColor}{plus petit commun multiple}  de $15$ et $20$. et on écrit $\ppcm(15, 20) = 60$}
\end{enumerate}
\end{Solution}



%% DEFINITION
\begin{Definition}[]
Soient $a$ et $b$ deux entiers naturels non nuls.

On appelle \textbf{ppcm} de $a$ et $b$ le \textbf{p}lus \textbf{p}etit \textbf{m}ultiple \textbf{c}ommun de $a$ et $b$, on le note souvent par
$ppcm(a, b)$ ou $a \vee b$.
\end{Definition}

%% APPLICATION
\begin{Application}
\begin{enumerate}
    \item Déterminer $\ppcm(12, 18)$.
    
    \item Déterminer $\ppcm(5;5)$ ; $\ppcm(1;3)$ ; $\ppcm(7;0)$ et $\ppcm(4;8)$.
\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][10]
\begin{enumerate}
    \item \textbf{Calcul de $\ppcm(12 \; ; \; 18)$ :}

    On décompose : $12 = 2^2 \times 3$ \quad et \quad $18 = 2 \times 3^2$

    Le $\ppcm$ s'obtient en prenant le produit des facteurs premiers avec le \textbf{plus grand exposant} :

    \[\ppcm(12 \; ; \; 18) = 2^2 \times 3^2 = 4 \times 9 = 36\]

    Par conséquent, $\ppcm(12 \; ; \; 18) = 36$.

    \item
    \begin{itemize}
        \item $\ppcm(5 \; ; \; 5) = 5$ \quad (le plus petit multiple commun de $5$ et $5$ est $5$).
        \item $\ppcm(1 \; ; \; 3) = 3$ \quad (tout multiple de $3$ est aussi un multiple de $1$).
        \item $\ppcm(7 \; ; \; 0) = 0$ \quad (par convention, $0$ est un multiple de tout entier, et le plus petit est $0$).
        \item $\ppcm(4 \; ; \; 8) = 8$ \quad (car $4 \mid 8$, donc tout multiple de $8$ est un multiple de $4$).
    \end{itemize}
\end{enumerate}
\end{Solution}

\section{Le plus grand commun diviseur \duree{15 min}}

%% ACTIVITE
\begin{Activite}
Un professeur d'EPS forme des équipes identiques (même nombre de filles et de garçons) avec $12$ filles et $18$ garçons. Tous les élèves participent.

\begin{enumerate}
    \item Peut-il former $3$ équipes ? Si oui, quelle sera la composition d'une équipe ?
    \item Peut-il former $4$ équipes ? Pourquoi ?
    \item Quel est le plus grand nombre d'équipes qu'il peut former ?
\end{enumerate}
\end{Activite}


%% SOLUTION_ACTIVITE
\begin{Solution}[ACTIVITÉ][5]
\begin{enumerate}
    \item On a : \quad    $12 \div 3 = 4   \quad \text{et}  \quad 18 \div 3 = 6$
        
    Puisque $4$ et $6$ sont des entiers naturels.
    
    Donc : le professeur peut former 3 équipes. Chaque équipe sera composée de 4 filles et 6 garçons.
    
    $\blacktriangleright$ \textbf{Ceci montre que 3 est un \textcolor{PrimaryColor}{diviseur commun} à 12 et 18.}
    
    \item On a : \quad $12 \div 4 = 3   \quad \text{et}  \quad     18 \div 4 = 4,5$
    
    Nous remarquons que $4,5$ n'est pas un entier (le nombre d'élèves doit être un entier).
    
    Donc : \quad il ne peut pas former 4 équipes, car 18 n'est pas divisible par 4.
    
    \item Cherchons les diviseurs de 12 et de 18 pour trouver le nombre maximum d'équipes possibles :
    
    On a : $D_{12} = \{1, 2, 3, 4, 6, 12\}$ \quad et \quad $D_{18} = \{1, 2, 3, 6, 9, 18\}$
    
    Donc les diviseurs communs à $12$ et $18$ sont $1, 2, 3 ~\text{et}~6$.
    
    D'où : le plus grand de ces diviseurs est $6$.
    
    Donc : le plus grand nombre d'équipes qu'il peut former est $6$.

    $\blacktriangleright$ \textbf{On dit que $6$ est le \textcolor{PrimaryColor}{plus grand commun diviseur}  de $12$ et $18$. et on écrit $\pgcd(12, 18) = 6$}
    

\end{enumerate}
\end{Solution}


%% DEFINITION
\begin{Definition}[]
Soient $a$ et $b$ deux entiers naturels non nuls.

On appelle \textbf{pgcd} de $a$ et $b$ le \textbf{p}lus \textbf{g}rand \textbf{c}ommun \textbf{d}iviseur  de $a$ et $b$, on le note souvent par
$pgcd(a, b)$ ou $a \wedge b$. 
\end{Definition}


%% APPLICATION
\begin{Application}
\begin{enumerate}
    \item Déterminer $\pgcd(40, 84)$.

    \item Déterminer $\pgcd(7;7)$ ; $\pgcd(1;3)$ ;  $\pgcd(7;0)$ et $\pgcd(4;8)$.

\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][11]
\begin{enumerate}
    \item \textbf{Calcul de $\pgcd(40 \; ; \; 84)$ :}

    On décompose : $40 = 2^3 \times 5$ \quad et \quad $84 = 2^2 \times 3 \times 7$

    Le $\pgcd$ s'obtient en prenant le produit des facteurs premiers \textbf{communs} avec le \textbf{plus petit exposant} :

    \[\pgcd(40 \; ; \; 84) = 2^2 = 4\]

    Par conséquent, $\pgcd(40 \; ; \; 84) = 4$.

    \item
    \begin{itemize}
        \item $\pgcd(7 \; ; \; 7) = 7$.
        \item $\pgcd(1 \; ; \; 3) = 1$ \quad (tout entier est divisible par $1$).
        \item $\pgcd(7 \; ; \; 0) = 7$ \quad (par convention, $\pgcd(a \; ; \; 0) = a$ car tout entier divise $0$).
        \item $\pgcd(4 \; ; \; 8) = 4$ \quad (car $4 \mid 8$).
    \end{itemize}
\end{enumerate}
\end{Solution}


%% DEFINITION
\begin{Definition}[Nombres premiers entre eux]
Soient $a$ et $b$ deux entiers naturels.

$a$ et $b$ sont \textbf{premiers entre eux} signifie que $\pgcd(a,b) = 1$,
c'est-à-dire leur seul diviseur commun est $1$.
\end{Definition}


%% EXEMPLE
\begin{Exemple}
\begin{enumerate}
    \item $5$ et $17$ sont premiers entre eux car $\pgcd(5; 17) = 1$.
    \item $1325$ et $2450$ ne sont pas premiers entre eux car ils sont tous les deux divisibles par $5$.
    \item Deux entiers consécutifs sont toujours premiers entre eux.
\end{enumerate}
\end{Exemple}


%% REMARQUE
\begin{Remarque}
Il ne faut pas confondre \textbf{nombre premier} et \textbf{nombres premiers entre eux}.

Par exemple, $15$ et $22$ sont premiers entre eux, mais ne sont pas des nombres premiers.
\end{Remarque}

\section{Application de la décomposition en produit de facteurs premiers \duree{15 min}}

%% PROPRIETE
\begin{Propriete}[]
Soient $a$ et $b$ deux entiers naturels non nuls.
\begin{itemize}
  \item On admet que le $\pgcd(a,b)$ est égal au produit des facteurs premiers \textbf{communs} qui ont \textbf{le plus petit exposant} dans les décompositions de $a$ et $b$.
  \item On admet que le $\ppcm(a,b)$ est égal au produit des facteurs premiers \textbf{communs et non communs} qui ont \textbf{le plus grand exposant} dans les décompositions de $a$ et $b$.
\end{itemize}
\end{Propriete}

%% EXEMPLE

    \item Déterminer le plus petit entier naturel non nul $m$ pour lequel $ma$ soit un carré parfait.
\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][12]
\begin{enumerate}
    \item \textbf{Décomposition en produit de facteurs premiers :}

    \begin{multicols}{2}
    \[
    \begin{array}{r|l}
    3960 & 2 \\
    1980 & 2 \\
    990  & 2 \\
    495  & 3 \\
    165  & 3 \\
    55   & 5 \\
    11   & 11 \\
    1    &
    \end{array}
    \]

    Donc $a = 2^3 \times 3^2 \times 5 \times 11$.

    \columnbreak

    \[
    \begin{array}{r|l}
    2100 & 2 \\
    1050 & 2 \\
    525  & 3 \\
    175  & 5 \\
    35   & 5 \\
    7    & 7 \\
    1    &
    \end{array}
    \]

    Donc $b = 2^2 \times 3 \times 5^2 \times 7$.
    \end{multicols}

    \item \textbf{Calcul de $\pgcd(a \; ; \; b)$ et $\ppcm(a \; ; \; b)$ :}

    On a : $a = 2^3 \times 3^2 \times 5^1 \times 11$ \quad et \quad $b = 2^2 \times 3^1 \times 5^2 \times 7$.

    \textbf{Calcul du $\pgcd(a \; ; \; b)$} — facteurs communs avec le plus petit exposant :

    \[\pgcd(a \; ; \; b) = 2^2 \times 3^1 \times 5^1 = 4 \times 3 \times 5 = 60\]

    Par conséquent, $\pgcd(a \; ; \; b) = 60$.

    \textbf{Calcul du $\ppcm(a \; ; \; b)$} — tous les facteurs avec le plus grand exposant :

    \[\ppcm(a \; ; \; b) = 2^3 \times 3^2 \times 5^2 \times 7 \times 11 = 8 \times 9 \times 25 \times 7 \times 11 = 138\,600\]

    Par conséquent, $\ppcm(a \; ; \; b) = 138\,600$.

    \item \textbf{Simplification de $\frac{a}{b}$ et $\sqrt{ab}$ :}

    On a :
    \[\frac{a}{b} = \frac{2^3 \times 3^2 \times 5 \times 11}{2^2 \times 3 \times 5^2 \times 7} = \frac{2 \times 3 \times 11}{5 \times 7} = \frac{66}{35}\]

    Par conséquent, $\frac{a}{b} = \frac{66}{35}$ (fraction irréductible car $\pgcd(66 \; ; \; 35) = 1$).

    Pour $\sqrt{ab}$ :

    On a :
    \begin{align*}
        ab &= (2^3 \times 3^2 \times 5 \times 11) \times (2^2 \times 3 \times 5^2 \times 7) = 2^5 \times 3^3 \times 5^3 \times 7 \times 11
    \end{align*}

    Donc :
    \begin{align*}
        \sqrt{ab} &= \sqrt{2^5 \times 3^3 \times 5^3 \times 7 \times 11} \\
                  &= 2^2 \times 3 \times 5 \times \sqrt{2 \times 3 \times 5 \times 7 \times 11} \\
                  &= 60\sqrt{2310}
    \end{align*}

    Par conséquent, $\sqrt{ab} = 60\sqrt{2310}$.

    \item \textbf{Plus petit entier naturel non nul $m$ pour que $ma$ soit un carré parfait :}

    On a $a = 2^3 \times 3^2 \times 5^1 \times 11^1$. Pour que $ma$ soit un carré parfait, tous les exposants doivent être pairs.

    \begin{itemize}
        \item Exposant de $2$ : $3$ (impair) $\Rightarrow$ il manque $2^1$.
        \item Exposant de $3$ : $2$ (pair) $\Rightarrow$ rien à ajouter.
        \item Exposant de $5$ : $1$ (impair) $\Rightarrow$ il manque $5^1$.
        \item Exposant de $11$ : $1$ (impair) $\Rightarrow$ il manque $11^1$.
    \end{itemize}

    Donc le plus petit $m$ est $m = 2 \times 5 \times 11 = 110$.

    Vérification : $ma = 2^4 \times 3^2 \times 5^2 \times 11^2 = (2^2 \times 3 \times 5 \times 11)^2 = 660^2$.

    Par conséquent, le plus petit entier naturel non nul est $m = 110$.
\end{enumerate}
