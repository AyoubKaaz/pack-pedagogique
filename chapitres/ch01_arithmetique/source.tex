%% SECTION
\section{Ensemble $\N$ --- Nombres pairs et nombres impairs \duree{1h15}}

%% ACTIVITE
\begin{Activite}[][10 min]
On considère la liste de nombres suivante : 
$-3 \quad ; \quad 14 \quad ; \quad 2,5 \quad ; \quad 0 \quad ; \quad 25 \quad ; \quad 36 \quad ; \quad -7 \quad ; \quad 41$
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item Parmi les nombres de cette liste, relever ceux qui sont des entiers positifs.

    (\textit{Ces nombres sont appelés des \textbf{entiers naturels}}).
    
    \item Parmi les entiers naturels trouvés, lesquels sont pairs ? impairs ?  
        
    \item Écrire chacun des entiers naturels pairs sous forme $2k$ avec $k$ un entier naturel.
    \item Écrire chacun des entiers naturels impairs sous forme $2k+1$ avec $k$ un entier naturel.
\end{enumerate}
\end{Activite}

%% SOLUTION_ACTIVITE
\begin{Solution}[ACTIVITÉ][1]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item Les entiers positifs sont $0 ~; 14 ~; 25 ~; 36$ et $41$.

    \begin{itemize}[label=$\blacktriangleright$, leftmargin=*, nosep]
    \item \textbf{Ces nombres constituent l'ensemble des \textcolor{PrimaryColor}{entiers naturels}, noté $\N$.}
    \end{itemize}

    \item Les nombres pairs sont $0 ; 14$ et $36$.
    
    Les nombres impairs sont $25$ et $41$.

    \begin{itemize}[label=$\blacktriangleright$, leftmargin=*, nosep]
    \item \textbf{Tout entier naturel est soit \textcolor{PrimaryColor}{pair}, soit \textcolor{PrimaryColor}{impair}.}
    \end{itemize}

    \item On a donc :
    \begin{itemize}[leftmargin=*,itemsep=2pt]
        \item $0 = 2 \times 0$ (avec $k=0$)

        \item $14 = 2 \times 7$ (avec $k=7$)

        \item $36 = 2 \times 18$ (avec $k=18$)
    \end{itemize}

    
    \begin{itemize}[label=$\blacktriangleright$, leftmargin=*, nosep]
    \item \textbf{Un entier naturel $n$ est \textcolor{PrimaryColor}{pair} s'il existe un entier naturel $k$ tel que $n = 2k$.}
    \end{itemize}
    
    \item On a donc :
    \begin{itemize}[leftmargin=*,itemsep=2pt]
        \item $25 = 24 + 1 = 2 \times 12 + 1$ (avec $k=12$)
        \item $41 = 40 + 1 = 2 \times 20 + 1$ (avec $k=20$)
    \end{itemize}

    \begin{itemize}[label=$\blacktriangleright$, leftmargin=*, nosep]
    \item \textbf{Un entier naturel $n$ est \textcolor{PrimaryColor}{impair} s'il existe un entier naturel $k$ tel que $n = 2k+1$.}
\end{itemize}
\end{enumerate}
\end{Solution}

%% DEFINITION
% durée estimée : 3 min
\begin{Definition}[]
\begin{itemize}[leftmargin=*,itemsep=2pt]
    \item On désigne par $\N$ l'\textbf{ensemble des entiers naturels}, et on écrit :
$\N = \{0\,;\,1\,;\,2\,;\,3\,;\,4\,;\ldots\}$
    
    \item On désigne par $\N^*$ l'\textbf{ensemble des entiers naturels non nuls}, et on écrit :
$\N^* = \{\,1\,;\,2\,;\,3\,;\,4\,;\ldots\}$
\end{itemize}
\end{Definition}



%% DEFINITION
% durée estimée : 2 min
\begin{Definition}[]
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Un \textbf{nombre pair} est un multiple de $2$.
  \item Un \textbf{nombre impair} est un nombre qui n'est pas pair.
\end{itemize}
\end{Definition}



%% REMARQUE
% durée estimée : 1 min
\begin{Remarque}
   Il existe une infinité d'entiers naturels.
\end{Remarque}

%% PROPRIETE
% durée estimée : 3 min
\begin{Propriete}[Écriture d'un nombre pair ou impair quelconque] Soit $n$ un entier naturel.
\begin{itemize}[leftmargin=*,itemsep=2pt]
    \item $n$ est \textbf{pair} si et seulement s'il s'écrit :
$n = 2k \quad \text{avec } k \in \N$

    \item $n$ est \textbf{impair} si et seulement s'il s'écrit :
$n = 2k + 1 \quad \text{avec } k \in \N$
\end{itemize}
\end{Propriete}

%% EXEMPLE
\begin{Exemple}[][5 min]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item $2022$ est un nombre pair, car $2022 = 2 \times 1011$
    \item $2023$ est un nombre impair, car $2023 = 2 \times 1011 + 1$
    \item Soit $n \in \N$, 
    On pose : $E = 4n + 11$ et $F = 2n + 6$
\begin{multicols}{2}
    \begin{itemize}[leftmargin=*,itemsep=2pt]
        \item
On a $E = 4n + 11 = 4n + 10 + 1 = 2(2n + 5) + 1$

On pose $K = 2n + 5 \in \N$

Donc $E = 2K + 1$ avec $K \in \N$

 D'où $E$ est un nombre impair.

        \columnbreak
        
        \item On a $F = 2n + 6 = 2(n + 3)$
    
    On pose $K = n + 3 \in \N$
    
    Donc $F = 2K$ avec $K \in \N$
    
    D'où $F$ est un nombre pair.
    \end{itemize}
\end{multicols}
\end{enumerate}
\end{Exemple}


%% APPLICATION
\begin{Application}[][15 min]
Soit $n$ un entier naturel non nul. 
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item Étudier la parité de chacun des nombres suivants :
    \begin{multicols}{4}
    \begin{itemize}[label=\textbullet]
        \item $A=8n+10$
        \item $B=8n-1$
        \item $C=n(2n+4)$
        \item $D=4n^2+6n+7$
    \end{itemize}
    \end{multicols}
    \item 
    \begin{enumerate}[leftmargin=*,itemsep=2pt]
        \item Montrer que $(n + 1)^2 - n^2$ est un nombre impair.
        \item Résoudre dans $\N$ l'équation : $(n + 1)^2 - n^2 = 2021$
    \end{enumerate}
\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][1]
Soit $n \in \N$.
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item
    \begin{itemize}[leftmargin=*,itemsep=2pt]
        \item \textbf{Parité de $A = 8n + 10$ :}

    On a : $A = 8n + 10 = 2(4n + 5)$

    Posons $k = 4n + 5 \in \N$

    Donc $A = 2k$ avec $k \in \N$.

    Par conséquent, $A$ est un nombre \textbf{pair}.

        \item \textbf{Parité de $B = 8n - 1$ :}

    On a : $B = 8n - 1 = 2(4n - 1) + 1$

    Posons $k = 4n - 1 \in \N$ (car $n \geq 1$)

    Donc $B = 2k + 1$ avec $k \in \N$.

    Par conséquent, $B$ est un nombre \textbf{impair}.

        \item \textbf{Parité de $C = n(2n + 4)$ :}

    On a : $C = n(2n + 4) = 2n(n + 2)$

    Posons $k = n(n + 2) \in \N$

    Donc $C = 2k$ avec $k \in \N$.

    Donc $C$ est un nombre \textbf{pair}.


        \item \textbf{Parité de $D = 4n^2 + 6n + 7$ :}

    On a :
    \begin{align*}
        D = 4n^2 + 6n + 7 
          = 4n^2 + 6n + 6 + 1 
          = 2(2n^2 + 3n + 3) + 1
    \end{align*}

    Posons $k = 2n^2 + 3n + 3 \in \N$

    Donc $D = 2k + 1$ avec $k \in \N$.

    Donc  $D$ est un nombre \textbf{impair}.
    \end{itemize}

    \item
    \begin{enumerate}[leftmargin=*,itemsep=2pt]
        \item \textbf{Montrons que $(n+1)^2 - n^2$ est impair :}

        Soit $n \in \N$,  on a :
        \begin{align*}
            (n+1)^2 - n^2 = n^2 + 2n + 1 - n^2
                          = 2n + 1
        \end{align*}

        Donc $(n+1)^2 - n^2$ est un nombre \textbf{impair}.

        \item \textbf{Résolution dans $\N$ de $(n+1)^2 - n^2 = 2021$ :}

        D'après la question \textbf{a)}, on a $(n+1)^2 - n^2 = 2n + 1$.

        Donc :
        \begin{align*}
            (n+1)^2 - n^2 = 2021 
            \text{ équivaut à } \quad & 2n + 1 = 2021 \\
            \text{ équivaut à } \quad & 2n = 2021 - 1 \\
            \text{ équivaut à } \quad & 2n = 2020 \\
            \text{ équivaut à } \quad & n = \frac{2020}{2} \\
            \text{ équivaut à } \quad & n = 1010
        \end{align*}
        Puisque $1010 \in \N$, \\
        Donc : \quad $S = \{1010\}$

    \end{enumerate}
\end{enumerate}
\end{Solution}

%% PROPRIETE
% durée estimée : 3 min
\begin{Propriete}[Opérations sur les nombres pairs et impairs]
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item La somme de deux nombres de \textbf{même parité} est \textbf{paire}.
  \item La somme de deux nombres de \textbf{parité différente} est \textbf{impaire}.
  \item Si l'un des facteurs d'un produit est pair, alors ce produit est \textbf{pair}.
  \item Tout entier naturel élevé à une puissance non nulle conserve sa parité.
\end{itemize}
\end{Propriete}

%% APPLICATION
\begin{Application}[][15 min]
Soient $x$  et $y$ deux entiers naturels, tels que $x$ est pair et $y$ est impair. Déterminer la parité des nombres suivants :
\begin{center}
\begin{multicols}{4}
\begin{itemize}[leftmargin=*,itemsep=2pt]
    \item $x + 8y$
    \item $(x + 1)(y + 1)$
    \item $3x^3 + y^2 + 11$
    \item $x^{2024} + xy + 17$
\end{itemize}
\end{multicols}
\end{center}
\end{Application}



%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][2]
Soient $x \in \N$ et $y \in \N$. \\
Puisque $x$ est pair et $y$ est impair, on applique les propriétés des opérations sur les nombres pairs et impairs :
\begin{itemize}[leftmargin=*,itemsep=2pt]
    \item Puisque $8$ est un nombre pair, alors le produit $8y$ est pair. \\
    Alors la somme $x + 8y$ est paire. \\
    Donc : \quad $x + 8y \text{ est un nombre pair.}$
    
    \item Puisque $x$ est pair et $1$ est impair, alors $x + 1$ est impair. \\
    Puisque $y$ est impair et $1$ est impair, alors $y + 1$ est pair. \\
    Donc le produit $(x + 1)(y + 1)$ est pair. 
    
    \item Puisque $x$ est pair, alors $x^3$ est pair. \\
    Donc le produit $3x^3$ est pair. \\
    Puisque $y$ est impair, alors $y^2$ est impair. \\
    Donc la somme $y^2 + 11$ est paire. \\
    D'où la somme $3x^3 + y^2 + 11$ est paire. 
    
    \item Puisque $x$ est pair, alors $x^{2024}$ est pair et le produit $xy$ est pair. \\
    Donc la somme $x^{2024} + xy$ est paire. \\
    Puisque $x^{2024} + xy$ est pair et $17$ est impair, alors la somme totale est impaire. \\
    Donc : \quad $x^{2024} + xy + 17 \text{ est un nombre impair.}$
\end{itemize}
\end{Solution}

%% PROPRIETE
% durée estimée : 2 min
\begin{Propriete}[Entiers consécutifs]
Soient $n$ et $n+1$ deux entiers consécutifs.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item La somme de deux entiers consécutifs est \textbf{impaire}.
  \item Le produit de deux entiers consécutifs est \textbf{pair}.
\end{itemize}
\end{Propriete}


%% APPLICATION
\begin{Application}[][15 min]
Soit $n$ un entier naturel.
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item Montrer que le nombre $n(n+1)$ est pair.
    \item Déterminer la parité des nombres suivants :
\begin{center}
$a=n^2+5n \quad ; \quad b=n^2+7n+13 \quad ; \quad c=n^4-n^2$
\end{center}
\item En déduire la valeur de $t$ :
\[  t=5\times(-1)^a-8\times(-1)^b+11\times(-1)^c \]
\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][3]
Soit $n \in \N$.
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item \textbf{Montrons que $n(n+1)$ est pair :}

    Parmi deux entiers consécutifs $n$ et $n+1$, l'un est pair et l'autre est impair.

    Donc leur produit contient un facteur pair.

    Par conséquent, $n(n+1)$ est un nombre \textbf{pair}.

    \item \textbf{Parité de $a$, $b$ et $c$ :}
    \begin{itemize}[leftmargin=*,itemsep=2pt]
        \item \textbf{Pour $a = n^2 + 5n$ :}
        
        On a : $a = n^2 + n + 4n = n(n+1) + 2(2n)$
        
        Puisque $n(n+1)$ est pair et $2(2n)$ est pair, leur somme est paire.
        
        Donc $a$ \text{ est pair}.

        \item \textbf{Pour $b = n^2 + 7n + 13$ :}
        
        On a : $b = n^2 + n + 6n + 12 + 1 = n(n+1) + 2(3n + 6) + 1$
        
        Puisque $n(n+1)$ est pair et $2(3n+6)$ est pair, leur somme est paire.
                
        Donc $b$ est impair.

        \item \textbf{Pour $c = n^4 - n^2$ :}
        
        On a :         
        \begin{align*}
c = n^2(n^2 - 1)  
 = n^2(n-1)(n+1) 
 = n(n-1) \times n(n+1)
\end{align*}
        
        Puisque $n(n-1)$ est le produit de deux entiers consécutifs, il est pair.
        
        D'où : $c$ est le produit d'un nombre pair par un autre entier.
        
        Donc $c$ est pair.
    \end{itemize}

    \item \textbf{Déduisons la valeur de $t$ :}
    
    On a : $t = 5 \times (-1)^a - 8 \times (-1)^b + 11 \times (-1)^c$
    
    Puisque $a$ est pair, $(-1)^a = 1$
    
    Puisque $b$ est impair, $(-1)^b = -1$
    
    Puisque $c$ est pair, $(-1)^c = 1$
    
    On a donc :
    $t = 5 \times 1 - 8 \times (-1) + 11 \times 1 
     = 5 + 8 + 11$
    
    D'où : \quad $t = 24$
\end{enumerate}
\end{Solution}

\section{Multiples et diviseurs d'un entier naturel \duree{1h}}

%% DEFINITION
% durée estimée : 3 min
\begin{Definition}[Multiples et diviseurs]
Soient $a$ et $b$ deux entiers naturels.

S'il existe un entier naturel $k$ tel que $a = k \times b$, on dit que :
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $b$ \textbf{divise} $a$ (ou $b$ est un \textbf{diviseur} de $a$),
  \item $a$ est \textbf{divisible} par $b$,
  \item $a$ est un \textbf{multiple} de $b$.
\end{itemize}
\end{Definition}

%% REMARQUE
% durée estimée : 2 min
\begin{Remarque}
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $0$ est un multiple de tous les entiers naturels.
  \item Tout entier naturel est divisible par $1$ et par lui-même.
\end{itemize}
\end{Remarque}

%% EXEMPLE
\begin{Exemple}[][5 min]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item On a $145 = 5 \times 29$ , donc $145$ est un multiple de $5$ et de $29$.
    
    \item Le nombre $40$ n'est pas divisible par $7$.
    
    \item L'ensemble des multiples de $30$ est : 
    $M_{30} = \{ 0 ~;~ 30 ~;~ 60 ~;~ 90 ~;~ \dots \}$ (il est infini).
    
    \item L'ensemble des diviseurs de $30$ est : $D_{30} = \{1~;~2~;~3~;~5~;~6~;~10~;~15~;~30\}$ (cet ensemble est fini).
\end{enumerate}
\end{Exemple}


%% APPLICATION
\begin{Application}[][15 min]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item Déterminer les multiples de $13$ compris entre $100$ et $199$.
    \item Montrer que $3333$ est un multiple de $33$.
\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][4]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item \textbf{Multiples de $13$ compris entre $100$ et $199$ :}

    Les multiples de $13$ s’écrivent sous la forme $13k$ avec $k \in \N$

    Cherchons donc les entiers naturels $k \in \N$ tels que $100 \leq 13k \leq 199$ :
    
    On a : $100 \leq 13k \leq 199 \quad \text{ équivaut à } \quad \frac{100}{13} \leq k \leq \frac{199}{13}$


    On a : $\frac{100}{13} \approx 7{,}69$ \quad et \quad $\frac{199}{13} \approx 15{,}30$

    Puisque $k$ est un entier naturel.

    Donc $k \in \{8 \; ; \; 9 \; ; \; 10 \; ; \; 11 \; ; \; 12 \; ; \; 13 \; ; \; 14 \; ; \; 15\}$.

    Les multiples de $13$ compris entre $100$ et $199$ sont donc :
    \[104 \; ; \; 117 \; ; \; 130 \; ; \; 143 \; ; \; 156 \; ; \; 169 \; ; \; 182 \; ; \; 195\]

    \item \textbf{Montrons que $3333$ est un multiple de $33$ :}
    
    On a :
    $3333 = 3300 + 33 = 33 \times 100 + 33 \times 1 = 33 \times (100 + 1)$
    
    Donc : $3333 = 33 \times 101$
        
    D'où : \quad $3333$ \text{ est un multiple de } $33$.
\end{enumerate}
\end{Solution}

%% PROPRIETE
\begin{Propriete}[Opérations sur les multiples]
Soient $a$, $b$ et $c$ des entiers naturels.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Si $a$ est multiple de $c$ et $b$ est multiple de $c$, alors $a + b$ est multiple de $c$ ;
        
        et si de plus $a \geq b$, alors $a - b$ est multiple de $c$.
  \item Si $a$ est multiple de $b$ et $b$ est multiple de $c$, alors $a$ est multiple de $c$.
  \item Si $c$ est multiple de $a$, alors $nc$ est multiple de $a$ pour tout $n \in \N$.
\end{itemize}
\end{Propriete}

%% PROPRIETE
\begin{Propriete}[Opérations sur les diviseurs]
Soient $a$, $b$ et $c$ des entiers naturels.
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item Si $a$ divise $b$ et $c$, et $b \geq c$, alors $a$ divise $b + c$ et $b - c$.
  \item Si $a$ divise $b$, alors $a$ divise $bc$.
\end{itemize}
\end{Propriete}

%% EXEMPLE
\begin{Exemple}[][5 min]
\begin{itemize}[leftmargin=*,itemsep=2pt]
    \item $20$ et $15$ sont des multiples de $5$, donc $20 + 15 = 35$ et $20 - 15 = 5$ sont aussi des multiples de $5$.
    \item $24$ est un multiple de $12$ et $12$ est un multiple de $3$, donc $24$ est un multiple de $3$.
    \item $8$ est un multiple de $4$, donc $8 \times 5 = 40$ est un multiple de $4$.
    \item $3$ divise $15$ et $9$, donc $3$ divise $15 + 9 = 24$ et $15 - 9 = 6$.
    \item $7$ divise $14$, donc $7$ divise $14 \times 2 = 28$.
\end{itemize}
\end{Exemple}

%% PROPRIETE
% durée estimée : 3 min
\begin{Propriete}[Critères de divisibilité] Un nombre est divisible par :
\begin{itemize}[leftmargin=*,itemsep=2pt]
  \item $2$ si son chiffre des unités est pair.
  \item $3$ si la somme de ses chiffres est divisible par $3$.
  \item $5$ si son chiffre des unités est $0$ ou $5$.
  \item $9$ si la somme de ses chiffres est divisible par $9$.
\end{itemize}
\end{Propriete}

%% APPLICATION
\begin{Application}[][15 min]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item Déterminer parmi les nombres suivants ceux qui sont divisibles à la fois par $2$ et par $3$ :
\begin{center}
    $126 \quad ; \quad 345 \quad ; \quad 810 \quad ; \quad 3054 \quad ; \quad 5051 \quad ; \quad 7204$
\end{center}
    \item Déterminer parmi les nombres suivants ceux qui sont divisibles à la fois par $5$ et par $9$ :
\begin{center}
    $315 \quad ; \quad 450 \quad ; \quad 333 \quad ; \quad 9045 \quad ; \quad 3800 \quad ; \quad 9003$
\end{center}
\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][5]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item Parmi les nombres proposés, ceux divisibles par $2$ sont : 
    
    $126$ , $810$  , $3054$ et $7204$.
    
    Cherchons parmi ces nombres ceux qui sont également divisibles par $3$ :
    \begin{itemize}[leftmargin=*,itemsep=2pt]
        \item On a $1+2+6 = 9$ , Donc $126$ est divisible par $3$.
        \item On a $8+1+0 = 9$ , Donc $810$ est divisible par $3$.
        \item On a $3+0+5+4 = 12$ , Donc $3054$ est divisible par $3$.
        \item On a $7+2+0+4 = 13$ , Donc $7204$ n'est pas divisible par $3$.
    \end{itemize}
    
    D'où : \quad les nombres divisibles à la fois par $2$ et par $3$ sont : $126$ , $810$ et $3054$.

    \item Parmi les nombres proposés, ceux divisibles par $5$ sont : $315$ , $450$ , $9045$ et $3800$.
    
    Cherchons parmi ces nombres ceux qui sont également divisibles par $9$ :
    \begin{itemize}[leftmargin=*,itemsep=2pt]
        \item Pour $315$ : on a $3+1+5 = 9$ , qui est divisible par $9$.
        \item Pour $450$ : on a $4+5+0 = 9$ , qui est divisible par $9$.
        \item Pour $9045$ : on a $9+0+4+5 = 18$ , qui est divisible par $9$.
        \item Pour $3800$ : on a $3+8+0+0 = 11$ , qui n'est pas divisible par $9$.
    \end{itemize}
    
   D'où : \quad les nombres divisibles à la fois par $5$ et par $9$ sont : $315$ , $450$ et $9045$.
\end{enumerate}
\end{Solution}



%% APPLICATION
\begin{Application}[][15 min]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item Déterminer toutes les valeurs possibles du chiffre $x$ pour que le nombre $N = \overline{4x3}$ soit divisible par $3$.
    \item Déterminer le chiffre $y$ pour que le nombre $M = \overline{5y}$ soit à la fois pair et divisible par $9$.
\end{enumerate}
\end{Application}



%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][6]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item \textbf{Valeurs de $x$ pour que $N = \overline{4x3}$ soit divisible par $3$ :}

    $N$ est divisible par $3$ si et seulement si la somme de ses chiffres est divisible par $3$.

    On a : $4 + x + 3 = 7 + x$

    Il faut donc que $3$ divise $(7 + x)$, avec $x \in \{0 \; ; \; 1 \; ; \; \ldots \; ; \; 9\}$.

    Puisque $0 \leq x \leq 9$, alors : $7 \leq 7 + x \leq 16$.
    
    Les multiples de $3$ compris entre $7$ et $16$ sont : $9$, $12$ et $15$.
    
    On a donc :
    \begin{align*}
        7 + x &= 9 \quad \text{ équivaut à } \quad x = 2 \\
        7 + x &= 12 \quad \text{ équivaut à } \quad x = 5 \\
        7 + x &= 15 \quad \text{ équivaut à } \quad x = 8
    \end{align*}
    
    Donc : \quad les valeurs possibles du chiffre $x$ sont $2$, $5$ et $8$.

    \item \textbf{Valeur de $y$ pour que $M = \overline{5y}$ soit pair et divisible par $9$ :}

    Puisque $M$ est pair, son chiffre des unités $y$ est pair.
    
    On a donc : $y \in \{0, 2, 4, 6, 8\}$.
    
    Puisque $M$ est divisible par $9$, la somme de ses chiffres est un multiple de $9$.
    
    On a donc : $5 + y$ est un multiple de $9$.
    
    Puisque $0 \leq y \leq 8$, alors : $5 \leq 5 + y \leq 13$.
    
    Le seul multiple de $9$ compris entre $5$ et $13$ est $9$.
    
    On a donc :
    \begin{align*}
        5 + y &= 9 \quad \text{ équivaut à } \quad y = 4
    \end{align*}
    
    Puisque $4$ est un chiffre pair, cette valeur convient.
    
    Par conséquent, $y = 4$.
\end{enumerate}
\end{Solution}

\end{enumerate}

%% SOLUTION_ACTIVITE
\begin{Solution}[ACTIVITÉ][2]

\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item Les diviseurs de chaque entier naturel :
\begin{multicols}{3}
\begin{itemize}[leftmargin=*,itemsep=2pt]
    \item $D_2 = \{1 ~;~ 2\}$
    \item $D_3 = \{1 ~;~ 3\}$
    \item $D_4 = \{1 ~;~ 2 ~;~ 4\}$
    \item $D_5 = \{1 ~;~ 5\}$
    \item $D_6 = \{1 ~;~ 2 ~;~  3 ~;~ 6\}$
    \item $D_7 = \{1 ~;~ 7\}$
    \item $D_{11} = \{1 ~;~ 11\}$
\end{itemize}
\end{multicols}

    \item Les nombres admettant exactement deux diviseurs sont : $2$, $3$, $5$, $7$ et $11$.
    
    \item \begin{itemize}[leftmargin=*,itemsep=2pt]
    \item $0$ n'est pas premier : tout entier naturel non nul divise $0$, donc $0$ a une infinité de diviseurs.
    \item $1$ n'est pas premier : il n'admet qu'un seul diviseur (lui-même).
\end{itemize}
    
    \item Le nombre $2$ est premier et pair. Tout autre nombre pair est divisible par $2$, donc admet au moins trois diviseurs ($1$, $2$ et lui-même). Par conséquent, $2$ est le seul nombre premier pair.
    
    \item On a : $1 + 0 + 7 + 1 = 9$, qui est divisible par $9$. Donc $9$ divise $1071$.

Par conséquent, $1071$ n'est pas un nombre premier.
\end{enumerate}

\end{Solution}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][7]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item \textbf{Montrons que $A = n^2 + n$ n'est pas premier (pour $n \geq 2$) :}

    On a : $A = n^2 + n = n(n + 1)$

    Or $n \geq 2$, donc $n \geq 2$ et $n + 1 \geq 3$.

    Ainsi, $A = n(n+1)$ est un produit de deux facteurs dont chacun est différent de $1$ et de $A$.

    Donc $A$ admet au moins quatre diviseurs distincts : $1$, $n$, $n+1$ et $A$.

    Par conséquent, $A$ n'est \textbf{pas un nombre premier}.

    \item \textbf{Justifions que $p + 7$ n'est pas premier :}

    On sait que $p$ est premier et $p > 2$, donc $p$ est impair (car $2$ est le seul nombre premier pair).

    Or $7$ est impair. Donc $p + 7$ est la somme de deux nombres impairs.

    Donc $p + 7$ est pair.

    De plus, $p \geq 3$, donc $p + 7 \geq 10 > 2$.

    Ainsi, le nombre $p + 7$ est un entier naturel strictement supérieur à $2$ qui est divisible par $2$.


    Par conséquent, $p + 7$ n'est \textbf{pas un nombre premier}.
\end{enumerate}
\end{Solution}

Pour que $n$ soit premier, il faut et il suffit qu'aucun nombre premier $p$ vérifiant $p \leq \sqrt{n}$ ne divise $n$.
\end{Theoreme}

Aucun de ces nombres ne divise $109$. Donc $109$ est premier.

\section{Le plus grand commun diviseur \duree{35 min}}

%% ACTIVITE
\begin{Activite}[][10 min]
Un professeur d'EPS forme des équipes identiques (même nombre de filles et de garçons) avec $12$ filles et $18$ garçons. Tous les élèves participent.

\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item Peut-il former $3$ équipes ? Si oui, quelle sera la composition d'une équipe ?
    \item Peut-il former $4$ équipes ? Pourquoi ?
    \item Quel est le plus grand nombre d'équipes qu'il peut former ?
\end{enumerate}
\end{Activite}


%% SOLUTION_ACTIVITE
\begin{Solution}[ACTIVITÉ][5]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item On a : \quad    $12 \div 3 = 4   \quad \text{et}  \quad 18 \div 3 = 6$

    Puisque $4$ et $6$ sont des entiers naturels.

    Donc : le professeur peut former 3 équipes. Chaque équipe sera composée de 4 filles et 6 garçons.

    \begin{itemize}[label=$\blacktriangleright$, leftmargin=*, nosep]
    \item \textbf{Ceci montre que 3 est un \textcolor{PrimaryColor}{diviseur commun} à 12 et 18.}
    \end{itemize}

    \item On a : \quad $12 \div 4 = 3   \quad \text{et}  \quad     18 \div 4 = 4,5$

    Nous remarquons que $4,5$ n'est pas un entier (le nombre d'élèves doit être un entier).

    Donc : \quad il ne peut pas former 4 équipes, car 18 n'est pas divisible par 4.

    \item Cherchons les diviseurs de 12 et de 18 pour trouver le nombre maximum d'équipes possibles :

    On a : $D_{12} = \{1~;~2~;~3~;~4~;~6~;~12\}$ \quad et \quad 
    $D_{18} = \{1~;~2~;~3~;~6~;~9~;~18\}$

    Donc les diviseurs communs à $12$ et $18$ sont $1, 2, 3 ~\text{et}~6$.

    D'où : le plus grand de ces diviseurs est $6$.

    Donc : le plus grand nombre d'équipes qu'il peut former est $6$.

    \begin{itemize}[label=$\blacktriangleright$, leftmargin=*, nosep]
    \item \textbf{On dit que $6$ est le \textcolor{PrimaryColor}{plus grand commun diviseur}  de $12$ et $18$. et on écrit $\pgcd(12, 18) = 6$}
    \end{itemize}

\end{enumerate}
\end{Solution}

%% DEFINITION
% durée estimée : 3 min
\begin{Definition}[]
Soient $a$ et $b$ deux entiers naturels non nuls.

On appelle \textbf{pgcd} de $a$ et $b$ le \textbf{p}lus \textbf{g}rand \textbf{c}ommun \textbf{d}iviseur  de $a$ et $b$, on le note souvent par
$\pgcd(a, b)$ ou $a \wedge b$.
\end{Definition}


%% APPLICATION
\begin{Application}[][10 min]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item Déterminer $\pgcd(40, 84)$.

    \item Déterminer $\pgcd(7;7)$, $\pgcd(1;3)$,  $\pgcd(7;0)$ et $\pgcd(4;8)$.

\end{enumerate}
\end{Application}

%% SOLUTION_APPLICATION
\begin{Solution}[APPLICATION][11]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item \textbf{Calcul de $\pgcd(40 \; ; \; 84)$ :}
    
    On a : l'ensemble des diviseurs de $40$ est :
    
    $D_{40} = \{1~;~2~;~4~;~5~;~8~;~10~;~20~;~40\}$.
    
    On a : l'ensemble des diviseurs de $84$ est :
    
    $D_{84} = \{1~;~2~;~3~;~4~;~6~;~7~;~12~;~14~;~21~;~28~;~42~;~84\}$.
    
    On a donc : l'ensemble des diviseurs communs de $40$ et $84$ est $\{1, 2, 4\}$.
    
    Puisque $4$ est le plus grand diviseur commun de ces deux nombres.
    
    Donc : \quad $\pgcd(40, 84) = 4$.

    \item
    \begin{itemize}[leftmargin=*,itemsep=2pt]
        \item $\pgcd(7 \; ; \; 7) = 7$ \quad
        (car pour tout entier naturel non nul $a$ : $\pgcd(a ; a) = a$)
        \item $\pgcd(1 \; ; \; 3) = 1$ \quad (tout entier est divisible par $1$).
        \item $\pgcd(7 \; ; \; 0) = 7$ \quad ( $\pgcd(a \; ; \; 0) = a$ car tout entier divise $0$).
        \item $\pgcd(4 \; ; \; 8) = 4$ \quad (car $4$ divise $8$).
    \end{itemize}
\end{enumerate}
\end{Solution}


%% DEFINITION
% durée estimée : 2 min
\begin{Definition}[Nombres premiers entre eux]
Soient $a$ et $b$ deux entiers naturels.

$a$ et $b$ sont \textbf{premiers entre eux} signifie que $\pgcd(a,b) = 1$,
c'est-à-dire leur seul diviseur commun est $1$.
\end{Definition}


%% EXEMPLE
\begin{Exemple}[][5 min]
\begin{enumerate}[leftmargin=*,itemsep=2pt]
    \item $5$ et $17$ sont premiers entre eux car $\pgcd(5; 17) = 1$.
    \item $1325$ et $2450$ ne sont pas premiers entre eux car ils sont tous les deux divisibles par $5$.
    \item Deux entiers consécutifs sont toujours premiers entre eux.
\end{enumerate}
\end{Exemple}


%% REMARQUE
% durée estimée : 2 min
\begin{Remarque}
Il ne faut pas confondre \textbf{nombre premier} et \textbf{nombres premiers entre eux}.

Par exemple, $15$ et $22$ sont premiers entre eux, mais ne sont pas des nombres premiers.
\end{Remarque}
