\begin{minipage}[t]{0.65\linewidth}
\begin{enumerate}
    \item Soit $f$ la fonction définie sur $\R$ par $f(x) = 3x^2$ et sa courbe $\mathcal{C}_f$ représentée ci-contre :
        \begin{enumerate}
            \item Vérifier que pour tout $x \in D_f$, $f(-x) = f(x)$
            \item Quelle est la propriété géométrique vérifiée par $\mathcal{C}_f$ ? 
        \end{enumerate}
        \vspace{1cm}
    \item Soit $g$ la fonction définie sur $\R^*$ par $g(x) = x^3$ et sa courbe $\mathcal{C}_g$ représentée ci-contre :
        \begin{enumerate}
            \item Vérifier que pour tout $x \in D_g$, $g(-x) = -g(x)$
            \item Quelle est la propriété géométrique vérifiée par $\mathcal{C}_g$ ? 
        \end{enumerate}
\end{enumerate}        
\end{minipage}
