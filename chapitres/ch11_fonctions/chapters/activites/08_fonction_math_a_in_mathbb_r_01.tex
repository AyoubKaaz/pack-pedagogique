\begin{Activite}[][10 min]
Soit $f$ la fonction définie sur $\mathbb{R}$ par : $f(x) = ax^2$ \quad ($a \in \mathbb{R}^*$)

On note $\big(\mathcal{C}_f\big)$ la courbe de $f$ dans un repère orthonormé $\big(O; \vect{i}; \vect{j}\big)$.
\begin{multicols}{2}
\begin{enumerate}
    \item Étudier la parité de $f$.
    \item Soient $x$ et $y$ deux réels distincts. 
    
    Montrer que : \quad $T(x;y) = a(x+y) $
    \item Étudier la monotonie de $f$ sur $[0 ; +\infty[$.
    \item En déduire la monotonie de $f$ sur $]-\infty ; 0]$.
    \item Dresser le tableau de variations de $f$ sur $\R$.
    \item On pose $a=1$.
    \begin{enumerate}
        \item Compléter le tableau suivant :
        \vspace{0.1cm}
         \begin{center}
            \renewcommand{\arraystretch}{1.2}
            \begin{tabular}{|Sc|*{6}{Sc|}}
                \hline
                $x$    & $-2$ & $-1$ & $0$ & $1$ & $2$  \\ \hline
                $f(x)$ &       &       &     &   &    \\ 
                \hline
            \end{tabular}
        \end{center}
        \vspace{0.1cm}
        \item Construire $\big(\mathcal{C}_f\big)$.
    \end{enumerate}
\end{enumerate}
\end{multicols}
\end{Activite}
