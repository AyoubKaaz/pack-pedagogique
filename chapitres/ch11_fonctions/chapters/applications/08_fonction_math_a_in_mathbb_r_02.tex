\begin{Application}[][10 min]
Soit $g$ la fonction définie sur $\mathbb{R}$ par : $g(x) = ax^2$  ~ ($a \in \mathbb{R}^*$)
\begin{enumerate}
    \item Déterminer la valeur de $a$ tel que $g(2) = -3$.
    \item Dresser le tableau de variations de $g$ sur $\mathbb{R}$.
    \item Tracer $(\mathcal{C}_g)$.
\end{enumerate}
\end{Application}
