\begin{Application}[][10 min]
Soit $f$ la fonction définie sur $\mathbb{R}$ par : $f(x) = 3x^2$
\begin{enumerate}
    \item Calculer $f(1)$ et $f(2)$ ; puis déduire $f(-1)$ et $f(-2)$.
    \item Dresser le tableau de variations de $f$ sur $\mathbb{R}$.
    \item Tracer $(\mathcal{C}_f)$.
\end{enumerate}
\end{Application}
