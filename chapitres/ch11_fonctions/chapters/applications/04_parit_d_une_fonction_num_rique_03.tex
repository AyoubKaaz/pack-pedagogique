\begin{Application}[][10 min]
\noindent
\begin{minipage}[t]{0.45\linewidth}
\vspace{0pt}
Soit $f$ une fonction paire définie sur $[-6 ; 6]$ par 

sa courbe $\mathcal{C}_f$ représentée ci-contre :
\begin{enumerate}
    \item Compléter la construction de $(C_f)$.
    \item Déterminer $f(6)$.
    \item Calculer $f(-3)$ et $f(3)$.
\end{enumerate}
\end{minipage}%
\hfill
\begin{minipage}[t]{0.55\linewidth}
\vspace{0pt}
\begin{center}
\begin{tikzpicture}[x=0.72cm, y=0.72cm]

    % 1. Grille de fond (couvrant toute la figure jusqu'aux extrémités des axes)
    \draw[black!12, step=1, solid] (-6.5,-2.5) grid (6.5,3.5);

    % 2. Axes principaux
    \draw[->, >=stealth, thick, black] (-6.5,0) -- (6.5,0)
        node[below right, font=\small\sffamily] {$x$};
    \draw[->, >=stealth, thick, black] (0,-2.5) -- (0,3.5)
        node[left, font=\small\sffamily] {$y$};

    % 3. Graduations axe X
    \foreach \x in {-6,-5,-4,-3,-2,-1,1,2,3} {
        \draw[PrimaryColor] (\x, 0.08) -- (\x, -0.08);
    }
    \node[below, font=\small\sffamily, color=black] at (-6, -0.15) {$-6$};
    \node[below, font=\small\sffamily, color=black] at (-5, -0.15) {$-5$};
    \node[above left, font=\small\sffamily, color=black] at (-2,  0.1) {$-2$};
    \node[above right, font=\small\sffamily, color=black] at ( 2,  0.1) {$2$};
    \node[below left, font=\small\sffamily, color=black] at (0, 0) {$O$};

    % 4. Graduations axe Y
    \foreach \y in {-1,1,2,3} {
        \draw[PrimaryColor] (0.08, \y) -- (-0.08, \y);
    }
    \node[right, font=\small\sffamily, color=black] at (0.12, -1) {$-1$};
    \node[right, font=\small\sffamily, color=black] at (0.12,  1) {$1$};
    \node[right, font=\small\sffamily, color=black] at (0.12,  2) {$2$};
    \node[right, font=\small\sffamily, color=black] at (0.12,  3) {$3$};

    % 5. (Lignes de rappel supprimées : la grille rend les coordonnées lisibles directement)

    % 6. Tracé de la fonction f(x)
    % Segment donné [-6, -2]
    \draw[very thick, SecondaryColor] (-6, 3) -- (-2, -1);
    % Segment donné [0, 2]
    \draw[very thick, SecondaryColor] (0, 2) -- (2, -1);

    % Points remarquables
    \filldraw[black] (-6,  3) circle (2pt);
    \filldraw[black] (-2, -1) circle (2pt);
    \filldraw[black] ( 0, -1) circle (2pt);
    \filldraw[black] ( 0,  2) circle (2pt);
    \filldraw[black] ( 2, -1) circle (2pt);

    % Label courbe
    \node[right, font=\small\sffamily\itshape, color=SecondaryColor]
        at (4, 2) {$(C_f)$};

\end{tikzpicture}
\end{center}
\end{minipage}
\end{Application}
