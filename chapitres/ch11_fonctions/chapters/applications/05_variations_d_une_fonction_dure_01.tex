\begin{Application}[][10 min]
Soit $f$ la fonction numérique définie sur $\mathbb{R}$ par : $f(x) = 2x^2 + 3$
\begin{enumerate}
    \item 
    \begin{enumerate}
        \item Soient $a$ et $b$ de $[0 ; +\infty[$ tels que $a < b$. Montrer que $f(a) < f(b)$
        \item En déduire la monotonie de $f$ sur $\mathbb{R}^+$.
    \end{enumerate}
    \item Étudier la monotonie de $f$ sur $\mathbb{R}^-$.
    \item Donner le tableau de variations de $f$ sur $\R$.
\end{enumerate}
\end{Application}
