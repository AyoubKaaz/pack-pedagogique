\begin{Application}[][10 min]
Soit $f$ la fonction définie sur $\mathbb{R}$ par : $f(x) = x^2 - 6x + 5$
\begin{enumerate}
    \item Soit $a$ et $b$ deux réels distincts. Montrer que $T(a;b) = a+b-6$
    \item Étudier la monotonie de $f$ sur chacun des deux intervalles $]-\infty ; 3]$ et $[3 ; +\infty[$.
    \item Dresser le tableau de variations de $f$.
    \item Comparer les nombres $f(2026)$ et $f(2027)$.
\end{enumerate}
\end{Application}
