\begin{Application}[][15 min]
Soit $f$ une fonction numérique définie par :
$$f(x) = x + \frac{4}{x}$$

\begin{enumerate}
    \item Déterminer $D_f$ ; puis montrer que $f$ est impaire.
    \item Soit $a$ et $b$ de $\R^*$ tel que $a \neq b$. 
    Montrer que : 
    $$\frac{f(b) - f(a)}{b - a} = \frac{ab - 4}{ab}$$
    \item Étudier les variations de $f$ sur chacun des intervalles $]0 ; 2]$ et $[2 ; +\infty[$.
    \item En déduire les variations de $f$ sur chacun des intervalles $]-\infty ; -2]$ et $[-2 ; 0[$.
    \item Dresser le tableau de variations de $f$.
\end{enumerate}
\end{Application}
