\begin{Exercice}
Soient $f$ et $g$ deux fonctions numériques définies par :
$$f(x) = -2x^2 + 4x \quad \text{et} \quad g(x) = \frac{x}{x - 1}$$

\begin{enumerate}
    \item Déterminer la nature de $(P)$ la courbe de $f$.
    \item Donner le tableau de variations de $f$.
    \item Construire $(C_f)$.
    \item Résoudre graphiquement dans $D_f$ : $f(x) = 3$ , $f(x) \geq 3$ et $f(x) \leq 3$.
    \item Donner le tableau de variations de $g$.
    \item Déterminer la nature de $(C_g)$ ; puis construire $(C_g)$.
    \item Résoudre graphiquement dans $D_g$ : $\frac{x}{x-1} = -2x^2 + 4x$ et $\frac{x}{x-1} \leq -2x^2 + 4x$.
\end{enumerate}
\end{Exercice}
