%% SECTION
\section{Fonction numérique d'une variable réelle \duree{15 min}}

%% ACTIVITE
\begin{Activite}[][5 min]
Ahmed dispose d'un morceau de fil de fer de $20\text{ cm}$ de longueur.
Il le plie pour former un rectangle de longueur $x$ et de largeur $y$.
\begin{enumerate}
    \item Exprimer la largeur $y$ en fonction de la longueur $x$.
    \item Déterminer l'intervalle $D$ auquel doit appartenir $x$ pour que ce rectangle puisse exister.
    \item Montrer que la surface du rectangle, notée $S(x)$, s'écrit $S(x) = 10x - x^2$
    \item Calculer la surface lorsque $x = 2$, puis lorsque $x = 8$.
    \item Pour quelle(s) valeur(s) de $x$ la surface est-elle égale à $25\text{ cm}^2$ ?
\end{enumerate}
\end{Activite}

%% DEFINITION
\begin{Definition}[Fonction numérique][]
\begin{itemize}
    \item On appelle \textbf{fonction numérique} d'une variable réelle $x$ toute relation $x \mapsto f(x)$ qui associe à chaque $x$ au plus une valeur.
    \item Si $f(x)$ existe, on dit que $f(x)$ est l'\textbf{image} de $x$.
    \item L'ensemble des nombres réels $x$ qui ont une image par $f$ est appelé \textbf{ensemble de définition} de $f$ et se note $D_f$.
\end{itemize}
\end{Definition}

%% EXEMPLE
\begin{Exemple}[][10 min]
\begin{enumerate}
    \item Soit la fonction numérique : $f(x) = \dfrac{2x}{x-1}$

    L'expression $\dfrac{2x}{x-1}$ est définie si : $x - 1 \neq 0$

    C'est-à-dire : $x \neq 1$ \\
    Donc l'ensemble de définition de $f$ est : $$D_f = ]-\infty ; 1[ \cup ]1 ; +\infty[$$ ou encore $D_f = \mathbb{R} \setminus \{1\}$

    \item Soit $g$ la fonction : $g(x) = \sqrt{2x - 3}$

    L'expression $\sqrt{2x - 3}$ est définie si $2x - 3 \geq 0$

    C'est-à-dire : $x \geq \dfrac{3}{2}$

    Donc l'ensemble de définition de $g$ est : $$D_g = \left[\dfrac{3}{2} ; +\infty\right[$$
\end{enumerate}
\end{Exemple}

%% APPLICATION
\begin{Application}[][10 min]
Déterminer l'ensemble de définition de la fonction $f$ dans chacun des cas suivants :
\begin{enumerate}
    \item $f(x) = x^2 - 2x + 1$
    \item $f(x) = \dfrac{x+1}{2x-8}$
    \item $f(x) = \sqrt{2-x}$
    \item $f(x) = \sqrt{x^2 - 5x + 6}$
    \item $f(x) = \dfrac{\sqrt{x}}{x^2 - 2x + 1}$
    \item $f(x) = \dfrac{x+4}{|x| - 3}$
    \item $f(x) = \dfrac{\sqrt{2-x}}{\sqrt{4x+2}}$
    \item $f(x) = \sqrt{\dfrac{2-x}{4x+2}}$
\end{enumerate}
\end{Application}

%% TECHNIQUES
\begin{Techniques}[][]
Soient $P(x)$ et $Q(x)$ deux polynômes ; on a :

\vspace{0.3cm}
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabular}{ScSc}
\toprule
\theadrow Fonction & Ensemble de définition \\
\midrule
$x \mapsto P(x)$ & $D_f = \mathbb{R} = ]-\infty ; +\infty[$ \\
$x \mapsto \dfrac{P(x)}{Q(x)}$ & $D_f = \{x \in \mathbb{R} / Q(x) \neq 0\}$ \\
$x \mapsto \sqrt{P(x)}$ & $D_f = \{x \in \mathbb{R} / P(x) \geq 0\}$ \\
$x \mapsto \dfrac{P(x)}{\sqrt{Q(x)}}$ & $D_f = \{x \in \mathbb{R} / Q(x) > 0\}$ \\
$x \mapsto \dfrac{\sqrt{P(x)}}{\sqrt{Q(x)}}$ & $D_f = \{x \in \mathbb{R} / P(x) \geq 0 \text{ et } Q(x) > 0\}$ \\
$x \mapsto \sqrt{\dfrac{P(x)}{Q(x)}}$ & $D_f = \left\{x \in \mathbb{R} / \dfrac{P(x)}{Q(x)} \geq 0 \text{ et } Q(x) \neq 0\right\}$ \\
\bottomrule
\end{tabular}
\end{center}
\end{Techniques}

\section{Égalité de deux fonctions \duree{15 min}}

%% DEFINITION
\begin{Definition}[hi][]
Soient $f$ et $g$ deux fonctions et $D_f$ et $D_g$ leurs ensembles de définition respectifs. \\
On dit que les fonctions $f$ et $g$ sont \textbf{égales } et on écrit $f = g$ si :
     $$D_f = D_g ~\text{ et pour tout}~  x ~\text{de}~ D_f : f(x) = g(x)$$
\end{Definition}

%% EXEMPLE
\begin{Exemple}[][10 min]
Soient $f$ et $g$ deux fonctions définies par :
 $f(x) = \sqrt{x^4 + 2x^2 + 1} \quad \text{et} \quad g(x) = x^2 + 1 $

\vspace{0.2cm}
On a $D_f = D_g = \mathbb{R}$ et
$f(x) = \sqrt{(x^2 + 1)^2} = x^2 + 1 \quad (\text{car } x^2 + 1 > 0)$ \\
Donc $f(x) = g(x)$, pour tout $x$ de $\mathbb{R}$.
D'où : \quad $f = g$
\end{Exemple}

%% APPLICATION
\begin{Application}[][5 min]
\noindent
On considère les fonctions numériques $f$ et $g$ définies par :
$f(x) = \frac{1}{\sqrt{x}-2} \quad \text{et} \quad g(x) = \frac{\sqrt{x}+2}{x-4}$

Montrer que $f=g$.
\end{Application}

\section{Représentation graphique d'une fonction \duree{15 min}}

%% DEFINITION
\begin{Definition}[hi][]
La \textbf{courbe représentative} de $f$ (ou \textbf{graphe} de $f$), notée $\mathcal{C}_f$, est l'ensemble des points $M(x\,;\,y)$ du plan tels que $x \in D_f$ et $y = f(x)$ :

$$\mathcal{C}_f = \bigl\{M\bigl(x\,;\,f(x)\bigr) \mid x \in D_f\bigr\}$$

            at (axis cs:2, 0) {$x$};
        \node[right, font=\small\sffamily\itshape, color=SecondaryColor]
            at (axis cs:3.7, 0.5) {$\mathcal{C}_f$};
    \end{axis}
\end{tikzpicture}
\end{center}

\end{Definition}

%% EXEMPLE
\begin{Exemple}[][10 min]
Soit $f$ la fonction définie sur $\mathbb{R}$ par : $f(x) = |2x - 1|$
\vspace{0.2cm}
\begin{itemize}
    \item On a $f\!\left(\frac{1}{2}\right) = 0$ donc le point $A\!\left(\frac{1}{2} ; 0\right) \in (\mathcal{C}_f)$
    \item On a $f(2) = 3$ donc le point $B(2 ; 3) \in (\mathcal{C}_f)$
\end{itemize}
\end{Exemple}

%% APPLICATION
\begin{Application}[][5 min]
\noindent
Considérons la fonction $f$ définie par $f(x) = \frac{x^2}{x+1}$

Parmi les points $A(0;0)$, $B(-1;1)$, $C\left(3;\frac{9}{4}\right)$ et $D(2;4)$

déterminer ceux qui appartiennent à $(C_f)$.
\end{Application}

%% DEFINITION
\begin{Definition}[][]
Soit $f$ une fonction et $D_f$ son ensemble de définition.
\begin{itemize}
    \item $f$ est \textbf{paire} si et seulement si : pour tout $x \in D_f$ : $-x \in D_f$ et $f(-x) = f(x)$
    \item $f$ est \textbf{impaire} si et seulement si : pour tout $x \in D_f$ : $-x \in D_f$ et $f(-x) = -f(x)$
\end{itemize}
\end{Definition}

%% REMARQUE
\begin{Remarque}[][]
 $x \in D_f$ et $-x \in D_f$ signifie que $D_f$ est symétrique par rapport à zéro.
\end{Remarque}

%% EXEMPLE
\begin{Exemple}[][10 min]
\begin{enumerate}
    \item Soit $f$ la fonction définie par :  $f(x) = x^4 - x^2$
    \vspace{0.1cm} \\
    On a $D_f = \mathbb{R}$. Il est clair que si $x \in \mathbb{R}$ alors $-x \in \mathbb{R}$ \\
    et on a $$f(-x) = (-x)^4 - (-x)^2 = x^4 - x^2$$
    Donc $f$ est paire.

    \item Soit $g$ la fonction définie par : $g(x) = x + x^3$
    \vspace{0.1cm} \\
    On a $D_g = \mathbb{R}$ est symétrique par rapport à $0$ \\
    et on a $$g(-x) = -x + (-x)^3 = -x - x^3 = -(x + x^3) = -g(x)$$
    Donc $g$ est une fonction impaire.
\end{enumerate}
\end{Exemple}

%% APPLICATION
\begin{Application}[][10 min]

    \begin{enumerate}
    \item $f(x) = x^2 - 3|x| + 2$
    \item $f(x) = 2x^3 - 5x$
    \item $f(x) = \sqrt{x - 1}$
\end{enumerate}
\end{Application}

%% APPLICATION
\begin{Application}[][10 min]

\begin{enumerate}
    \item Calculer $f(1)$ et $f(-1)$.
    \item En déduire que $f$ n'est ni paire, ni impaire.
\end{enumerate}
\end{Application}

\begin{Propriete}[][5 min]
Soit $f$ une fonction et $(C_f)$ sa courbe dans un repère orthogonal $(O ; \vec{i} ; \vec{j})$.

\end{Propriete}

%% DEFINITION
\begin{Definition}[hi][]
Soit $f$ une fonction numérique et $I$ un intervalle inclus dans $D_f$.
\begin{itemize}
    \item $f$ est \textbf{strictement croissante} (resp. croissante) sur $I$ si pour tous réels $a$ et $b$ de $I$ tels que : $a < b$ on a $f(a) < f(b)$ (resp. $f(a) \leq f(b)$).
    \item $f$ est \textbf{strictement décroissante} (resp. décroissante) sur $I$ si pour tous réels $a$ et $b$ de $I$ tels que $a < b$ on a : $f(a) > f(b)$ (resp. $f(a) \geq f(b)$).
    \item $f$ est \textbf{constante} sur $I$ si pour tous réels $a$ et $b$ de $I$ tels que $a < b$ on a $f(a) = f(b)$.
    \item $f$ est \textbf{strictement monotone} (resp. monotone) sur $I$ si elle est strictement croissante (resp. croissante) ou strictement décroissante (resp. décroissante) sur $I$.
\end{itemize}
\end{Definition}

%% SECTION
\subsection{Taux de variations d'une fonction \duree{15 min}}

%% DEFINITION
\begin{Definition}[hi][]
Soit $f$ une fonction, et soit $a$ et $b$ deux éléments distincts de $D_f$. \\
Le nombre réel $T(a;b) = \frac{f(a)-f(b)}{a-b}$ est appelé \textbf{taux de variations} de $f$ entre $a$ et $b$.
\end{Definition}

%% EXEMPLE
\begin{Exemple}[][10 min]
Soit $f$ la fonction définie par $f(x) = -5x + 3 \quad$ et $\quad D_f = \mathbb{R}$.
\vspace{0.1cm} \\
Soit $a$ et $b$ deux éléments de $\mathbb{R}$ tels que $a \neq b$. On a :
\[ T(a;b) = \frac{f(a)-f(b)}{a-b} = \frac{(-5a+3) - (-5b+3)}{a-b} = \frac{-5(a-b)}{a-b} = -5 \]
\end{Exemple}

%% PROPRIETE
\begin{Propriete}[][]
Soit $f$ une fonction et $T$ son taux de variation entre deux réels distincts $a$ et $b$ d'un intervalle $I$ inclus dans $D_f$.
\begin{itemize}
    \item Si $T > 0$ (resp. $T \geq 0$) pour tous $a$ et $b$ de $I$, alors $f$ est strictement croissante (resp. croissante) sur $I$.
    \item Si $T < 0$ (resp. $T \leq 0$) pour tous $a$ et $b$ de $I$, alors $f$ est strictement décroissante (resp. décroissante) sur $I$.
\end{itemize}
\end{Propriete}

%% EXEMPLE
\begin{Exemple}[][10 min]
Soit $f$ la fonction numérique définie par : $f(x) = x^2 + x$
\vspace{0.1cm} \\
Soit $a$ et $b$ deux réels distincts de $\R$. On a  :
\[ T(a ; b) = \frac{f(a) - f(b)}{a - b}  = \frac{(a^2 + a) - (b^2 + b)}{a - b} = \frac{a^2 - b^2 + a - b}{a - b}
 = \frac{(a - b)(a + b) + (a - b)}{a - b}
 = \frac{(a - b)(a + b + 1)}{a - b} \]
Puisque $a \neq b$, on a $a - b \neq 0$, on peut donc simplifier :
 $T(a ; b) = a + b + 1$

\medskip

\textbf{La monotonie sur l'intervalle } $\left]-\infty ; -\frac{1}{2}\right]$ \textbf{:}

Soit $a$ et $b$ deux réels distincts de $\left]-\infty ; -\frac{1}{2}\right]$,

On a :~~ $a \leq -\frac{1}{2}$ ~et~ $b < -\frac{1}{2}$  (Car $a \neq b$)

Donc : \qquad  $a + b < -1$

C'est-à-dire :  $a + b + 1 < 0$

Donc : $T(a ; b) < 0$, et par suite :

$f$ est strictement décroissante sur $\left]-\infty ; -\frac{1}{2}\right]$.

\medskip

\textbf{La monotonie sur l'intervalle } $\left[-\frac{1}{2} ; +\infty\right[$ \textbf{:}

Soit $a$ et $b$ deux réels distincts de $\left[-\frac{1}{2} ; +\infty\right[$.

On a $a \geq -\frac{1}{2}$ et $b > -\frac{1}{2}$ \quad (Car $a \neq b$)

Donc :~~~ $a + b > -1$

C'est-à-dire : ~ $a + b + 1 > 0$

Donc : $T(a ; b) > 0$, et par suite :

$f$ est strictement croissante sur $\left[-\frac{1}{2} ; +\infty\right[$.

Ces résultats peut être résumés dans un tableau, appelé \textcolor{purple}{Tableau de variations} de la fonction $f$.

\end{Exemple}

\subsection{Monotonie et parité d'une fonction \duree{15 min}}

%% PROPRIETE
\begin{Propriete}[][5 min]
Soit $f$ une fonction dont $D_f$ est symétrique par rapport à $0$. \\
Soit $I$ un intervalle de $\mathbb{R}$ tel que $I \subset D_f$ et $I'$ le symétrique de $I$ par rapport à $0$.
\begin{enumerate}
    \item[(1)] Dans le cas où $f$ est \textbf{paire}, on a :
    \begin{itemize}
        \item Si $f$ est croissante sur $I$, alors $f$ est décroissante sur $I'$.
        \item Si $f$ est décroissante sur $I$, alors $f$ est croissante sur $I'$.
    \end{itemize}
    \item[(2)] Dans le cas où $f$ est \textbf{impaire}, alors $f$ a le même sens de variation sur $I$ et $I'$.
\end{enumerate}
\end{Propriete}

%% EXEMPLE
\begin{Exemple}[][10 min]
\begin{itemize}
    \item La fonction $f : x \mapsto |x|$ est paire ; elle est croissante sur $I = [0 ; +\infty[$ et décroissante sur $I' = ]-\infty ; 0]$.
    \item La fonction $g : x \mapsto \frac{1}{x}$ est impaire, elle est décroissante sur chacun des intervalles $]-\infty ; 0[$ et $]0 ; +\infty[$.
\end{itemize}
\end{Exemple}

\section{Extremums d'une fonction numérique \duree{10 min}}

%% ACTIVITE
\begin{Activite}[][5 min]
Un joueur de foot tire un coup franc. La hauteur du ballon (exprimée en mètres) par rapport au sol en fonction du temps $t$ (exprimé en secondes) est donnée par la fonction $h$ définie sur l'intervalle $[0; 6]$ par :
$$h(t) = -t^2 + 6t$$

\begin{enumerate}
    \item Calculer $h(0)$ et $h(6)$. Que signifient ces deux résultats pour le ballon ?
    \item Calculer la hauteur du ballon après $3$ secondes.
    \item
    \begin{enumerate}
    \item Montrer que pour tout $t \in [0; 6]$, on a : $h(t) - 9 = -(t - 3)^2$
    \item En déduire que pour tout $t \in [0; 6]$, on a : $h(t) \leq 9$
    \end{enumerate}
    \item Quelle est la hauteur maximale atteinte par le ballon ? À quel instant est-elle atteinte ?
\end{enumerate}
\end{Activite}

%% DEFINITION
\begin{Definition}[][]
Soit $f$ une fonction numérique ; $I$ un intervalle inclus dans $D_f$ et $a \in I$.
\begin{itemize}
    \item On dit que $f(a)$ est le \textbf{maximum} de $f$ sur $I$ si :
    $f(x) \leq f(a)$ pour tout $x$ de $I$.
    \item On dit que $f(a)$ est le \textbf{minimum} de $f$ sur $I$ si :
    $f(x) \geq f(a)$ pour tout $x$ de $I$.
\end{itemize}
\end{Definition}

%% APPLICATION
\begin{Application}[][10 min]
On considère la fonction $f(x) = x^2 - 6x + 7$
\begin{enumerate}
    \item Vérifier que $f(x) \geq -2$ pour tout $x$ de $\mathbb{R}$.
    \item Calculer $f(3)$, puis en déduire que $-2$ est le minimum de $f$ sur $\mathbb{R}$.
\end{enumerate}
\end{Application}

%% APPLICATION

    \item Calculer $f(1)$, puis en déduire que $5$ est le maximum de $f$ sur $\mathbb{R}$.
\end{enumerate}
\end{Application}

%% REMARQUE
\begin{Remarque}[][]
\begin{itemize}
    \item $M$ est le maximum de $f$ sur $I$ si et seulement si :
    \begin{enumerate}
        \item Pour tout $x$ de $I$ : $f(x) \leq M$
        \item L'équation $f(x) = M$ admet au moins une solution dans $I$.
    \end{enumerate}
    \item $m$ est le minimum de $f$ sur $I$ si et seulement si :
    \begin{enumerate}
        \item Pour tout $x$ de $I$ : $f(x) \geq m$
        \item L'équation $f(x) = m$ admet au moins une solution dans $I$.
    \end{enumerate}
\end{itemize}
\end{Remarque}

\section{Fonctions périodiques \duree{15 min}}

%% DEFINITION
\begin{Definition}[][]
Soit $f$ une fonction numérique et $D_f$ son ensemble de définition. \\
On dit que $f$ est \textbf{périodique} s'il existe un réel $T$ strictement positif tel que :
\begin{itemize}
    \item $x + T \in D_f$ ; pour tout $x$ de $D_f$.
    \item $f(x + T) = f(x)$ ; pour tout $x$ de $D_f$.
\end{itemize}

On dit que $T$ est \textbf{une période} de $f$ et que $f$ est \textbf{$T$-périodique}.
\end{Definition}

\section{La fonction homographique $x \mapsto \dfrac{ax+b}{cx+d}$ ; ($c \neq 0$ et $ad-bc \neq 0$)\duree{25 min}}

%% DEFINITION
\begin{Definition}
\begin{itemize}
    \item Toute fonction $f$ qui peut s'écrire sous la forme $f(x) = \frac{ax+b}{cx+d}$ où $a, b, c, d$ sont des réels tels que $c \neq 0$ et $ad-bc \neq 0$ est appelée fonction \textbf{homographique}.
    \item On note $\Delta$ le \textbf{déterminant} de $f$ et on a : $\Delta = \begin{vmatrix} a & b \\ c & d \end{vmatrix} = ad - bc$
\end{itemize}
\end{Definition}

%% EXEMPLE
\begin{Exemple}
\begin{enumerate}
    \item Soit $f$ la fonction définie par : $f(x) = \frac{2x+1}{x-3}$

    On a $\Delta = \begin{vmatrix} 2 & 1 \\ 1 & -3 \end{vmatrix} = 2(-3) - 1(1) = -7 \neq 0$

    Donc $f$ est une fonction homographique.

    \item Soit $g$ la fonction définie par : $g(x) = \frac{x}{x-1}$

    On a  $\Delta = \begin{vmatrix} 1 & 0 \\ 1 & -1 \end{vmatrix} = 1(-1) - 1(0) = -1 \neq 0$

    Donc $g$ est une fonction homographique.
\end{enumerate}
\end{Exemple}
