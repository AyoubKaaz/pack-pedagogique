    \item Compléter par  $\in$ ou $\notin$ :
    $$15 \dots \N \quad ; \quad -8 \dots \N \quad ; \quad -8 \dots \Z \quad ; \quad \frac{1}{3} \dots \D \quad ; \quad \sqrt{7} \dots \Q \quad ; \quad \pi \dots \R$$

    \item Compléter par  $\subset$ ou  $\not\subset$ :
    $$\N \dots \Z \quad ; \quad \R \dots \Q \quad ; \quad \D \dots \R \quad ; \quad \Z \dots \N \quad ; \quad 
    \Z \dots \R^+    $$

\item Simplifier les nombres suivants si possible, puis déterminer le plus petit ensemble auquel appartient chacun d'eux, en justifiant :
$$ -7 ~;~ 3{,}14 ~;~ \frac{1}{3} ~;~ \pi ~;~ \frac{12}{11} ~;~ \sqrt{1{,}44} ~;~ \frac{-\sqrt{3}}{4} $$
\end{enumerate}
\end{Application}

%% ACTIVITE
\begin{Activite}[][20 min]
Soit $a$ et $b$ deux réels.
\begin{enumerate}
    \item En écrivant $(a+b)^3 = (a+b)^2 \times (a+b)$, développer $(a+b)^3$.
    
    \item En remplaçant $b$ par $-b$ dans le résultat de la question \textbf{1.}, déduire le développement de $(a-b)^3$.
\end{enumerate}

\begin{enumerate}
    \setcounter{enumi}{2}
    \item Développer et réduire le produit $(a-b)\big(a^2 + ab + b^2\big)$.
    
    \item Développer et réduire le produit $(a+b)\big(a^2 - ab + b^2\big)$.
    
    \item À l'aide des questions \textbf{3.} et \textbf{4.}, compléter :
    \[
        a^3 - b^3 = \dots \qquad ; \qquad a^3 + b^3 = \dots
    \]
\end{enumerate}
\end{Activite}

%% EXERCICE
\begin{Exercice}[][30 min]
\begin{enumerate}

\item  Soit $x$ un réel. On considère les deux expressions suivantes :
$$E(x) = x^3 - 8 - 3(x - 2)(x + 1) \quad \text{et} \quad
F(x) = (x + 3)^3 - x(x - 2)^2$$
\begin{enumerate}
    \item Développer et réduire $F(x)$.
    \item Factoriser l'expression $x^3 - 8$.
    \item En déduire une factorisation de $E(x)$.
    \item Développer et réduire $E(x)$.
    \item Calculer la valeur exacte de $E(\sqrt{2})$.
\end{enumerate}

\item Soit le nombre réel $X = \sqrt{7 + 4\sqrt{3}} + \sqrt{7 - 4\sqrt{3}}$.

\begin{enumerate}
    \item Calculer $X^2$.
    \item Déterminer le signe de $X$, puis en déduire sa valeur exacte.
\end{enumerate}


\item Soient $x$ et $y$ deux réels non nuls. On pose : 
$M = \frac{x^{-2} y^3 \big(x^2 y^{-1}\big)^2}{x^3 y^{-2}}$

\begin{enumerate}
    \item Simplifier $M$.
    \item On donne $x = 3 \times 10^2$ et $y = 3 \times 10^{-1}$. Calculer $M$ et donner son écriture scientifique.
\end{enumerate}
\end{enumerate}
\end{Exercice}
