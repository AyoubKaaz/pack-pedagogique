\begin{Solution}[APPLICATION][2]
\begin{enumerate}[leftmargin=*,itemsep=0.5pt]
    \item \textbf{DĂŠveloppement et rĂŠduction :}
    \begin{itemize}[leftmargin=*,itemsep=0.5pt]
        \item $A = (2x + 3)(x - 4) + 5x$ \\
        $ A = 2x^2 - 8x + 3x - 12 + 5x$ \\
        $ A = 2x^2 - 12$
        
        \item $B = 3x(x - 2) - (x - 1)(x + 5)$ \\
        $B = (3x^2 - 6x) - (x^2 + 5x - x - 5)$ \\
        $B = 3x^2 - 6x - (x^2 + 4x - 5)$ \\
        $B = 3x^2 - 6x - x^2 - 4x + 5$ \\
        $B = 2x^2 - 10x + 5$
        
        \item $C = \sqrt{2}(x\sqrt{2} - 3) - x(2 - \sqrt{2}x)$ \\
        $C = (\sqrt{2} \times x\sqrt{2}) - 3\sqrt{2} - 2x + \sqrt{2}x^2$ \\
        $C = 2x - 3\sqrt{2} - 2x + \sqrt{2}x^2$ \\
        $C = \sqrt{2}x^2 - 3\sqrt{2}$
    \end{itemize}

    \item \textbf{Factorisation :}
    \begin{itemize}[leftmargin=*,itemsep=0.5pt]
        \item $A = 7x^2 - 21x$ \\
        $A = 7x \times x - 7x \times 3$ \\
        $A = 7x(x - 3)$
        
        \item $B = (3x - 1)(x + 2) - (3x - 1)(2x - 5)$ \\
        $B = (3x - 1) \left[ (x + 2) - (2x - 5) \right]$ \\
        $B = (3x - 1)(x + 2 - 2x + 5)$ \\
        $B = (3x - 1)(-x + 7)$
        
        \item $C = \sqrt{12}x^2 - \sqrt{3}x$ \\
        On sait que $\sqrt{12} = \sqrt{4 \times 3} = 2\sqrt{3}$. \\
        $C = 2\sqrt{3}x^2 - \sqrt{3}x$ \\
        $C = \sqrt{3}x(2x - 1)$
    \end{itemize}
\end{enumerate}
\end{Solution}

%% SOLUTION_ACTIVITE
\begin{Solution}[ACTIVITĂ][2]
\begin{enumerate}
    \item On a :
    \[
        (a+b)^3 = (a+b)^2(a+b) = \big(a^2 + 2ab + b^2\big)(a+b)
    \]
    En distribuant :
    \[
     (a+b)^3   = a^3 + a^2 b + 2a^2 b + 2ab^2 + ab^2 + b^3 = a^3 + 3a^2 b + 3ab^2 + b^3
    \]

    \item On remplace $b$ par $-b$ :
    \[
        (a-b)^3 = a^3 + 3a^2(-b) + 3a(-b)^2 + (-b)^3 = a^3 - 3a^2 b + 3ab^2 - b^3
    \]

    \item On dĂŠveloppe :
    \[
        (a-b)\big(a^2 + ab + b^2\big) = a^3 + a^2 b + ab^2 - a^2 b - ab^2 - b^3 = a^3 - b^3
    \]

    \item On dĂŠveloppe :
    \[
        (a+b)\big(a^2 - ab + b^2\big) = a^3 - a^2 b + ab^2 + a^2 b - ab^2 + b^3 = a^3 + b^3
    \]

    \item On en dĂŠduit :
    \[
        a^3 - b^3 = (a-b)\big(a^2 + ab + b^2\big) \qquad ; \qquad a^3 + b^3 = (a+b)\big(a^2 - ab + b^2\big)
    \]
\end{enumerate}
\end{Solution}
