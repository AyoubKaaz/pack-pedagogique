%% EXO_PROGRESSIF
\begin{Exercice}[][20 min][★★]
\begin{enumerate}
    \item Montrer que $\dfrac{2^n}{5^m} \in \D$ pour tout $n$ et $m$ de $\N$.
    \item Soit $n$ un entier naturel. Montrer que :
        $\dfrac{\big(9^{n+1}+9^n\big)^2}{\big(3^{2n+1}-3^{2n}\big)^2} \in \N$

    \item Montrer que $\dfrac{1}{3} \notin \D$.
    
\end{enumerate}
\end{Exercice}
%% EXO_PROGRESSIF

%% EXO_PROGRESSIF
\begin{Exercice}[][20 min][★★★]
\begin{enumerate}
    \item Soit $n$ un entier naturel non nul.
    
Le nombre $\sqrt{n^2 + n}$ peut-il appartenir à $\mathbb{N}$ ? Justifier.

    \item Soit $n$ un entier naturel non nul. On pose $A = \frac{10^n + 3}{10^n}$.

    Montrer que $A \notin \N$.
    
\end{enumerate}
\end{Exercice}

\begin{Exercice}[][20 min][★★★]
Soient $a$ et $b$ deux nombres réels tels que :
\begin{equation*}
    a+b=3   ~~~~\text{et}~~~~  a^2+b^2=11
\end{equation*}
\begin{enumerate}
    \item Montrer que : $ab=-1$
    \item Calculer : $a^3+b^3~~;~~a^4+b^4~$ ~~et~~ $a^6+b^6$
\end{enumerate}
\end{Exercice}

    \item En utilisant la relation précédente, calculer $A$ et $B$ où :

\end{enumerate}
\end{Exercice}

\begin{Exercice}[][20 min][★★]
Soient $a$ et $b$ deux réels non nuls distincts, et $n$ de $\Z$.

On pose : $B = \dfrac{a^{2n} - a^nb^n}{b^{2n} - a^nb^n} \times \left(\dfrac{b}{a}\right)^n$ ~et~~ $C = \left(\dfrac{a^{n+1} - a^nb}{b^{n+1} - ab^n}\right)^2 \times \left(\dfrac{b}{a}\right)^{2n}$
\vspace{0.2cm}\\
Montrer que $B=-1$ et $C=1$.
\end{Exercice}

\begin{Exercice}[][20 min][★★]
Soit $a$ et $b$ deux réels tels que :
\begin{equation*}
    a=1+\sqrt{4-\frac{3}{2}\sqrt{7}} ~~~\text{et}~~~ b=1+\sqrt{4+\frac{3}{2}\sqrt{7}}
\end{equation*}
\begin{enumerate}
    \item Vérifier que :
    \vspace{0.2cm} \\
    $(3-\sqrt{7})^2=4\left(4-\frac{3}{2}\sqrt{7}\right)$
    ~~~\text{et}~~~ $(3+\sqrt{7})^2=4\left(4+\frac{3}{2}\sqrt{7}\right)$
    \item Montrer que : $a+b=5$
    \item Montrer que : $(a-1)(b-1)=\frac{1}{2}$
    \item En déduire la valeur du produit $ab$.
\end{enumerate}
\end{Exercice}

\begin{Exercice}[][20 min][★★]
Soit $x$ un nombre réel tel que :~~ $x \ge 1 \quad \text{et} \quad x^2 + \dfrac{1}{x^2} = 7$
\begin{enumerate}
    \item Montrer que ~ $x + \dfrac{1}{x} = 3$

    \item Justifier que $x - \dfrac{1}{x} \ge 0$ puis déduire que
    $x - \dfrac{1}{x} = \sqrt{5}$

    \item En déduire la valeur exacte de $x$.

    \item Montrer que $x^3 + \frac{1}{x^3} = 18$

    \item Déterminer la valeur exacte de $x^2 - \dfrac{1}{x^2}$
\end{enumerate}
\end{Exercice}

\begin{Exercice}[][20 min][★★★]
Soient $x$ et $y$ deux réels strictement positifs tels que $x > y$ et :
$$ \sqrt{\frac{x}{y}} - \sqrt{\frac{y}{x}} = \sqrt{2} $$
\begin{enumerate}
    \item Montrer que $\frac{x}{y} + \frac{y}{x} = 4$

    \item En déduire que $x^2 + y^2 = 4xy$

    \item Calculer la valeur exacte de l'expression $A = \frac{x+y}{x-y}$

    \item On suppose que $xy = 2$. Déduire les valeurs de $x$ et $y$.
\end{enumerate}
\end{Exercice}
