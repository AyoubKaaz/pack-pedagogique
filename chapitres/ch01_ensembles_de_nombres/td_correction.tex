\begin{Solution}[APPLICATION][1]
\begin{enumerate}
    \item Complétons par $\in$ ou $\notin$ :
    
    $15 \in \N \quad ; \quad -8 \notin \N \quad ; \quad -8 \in \Z \quad ; ~ \frac{1}{3} \notin \D \quad ; ~ \sqrt{7} \notin \Q \quad ; \quad \pi \in \R$

    \item Complétons par $\subset$ ou $\not\subset$ :
    
    $\N \subset \Z \quad ; \quad \R \not\subset \Q \quad ; \quad \D \subset \R \quad ; \quad \Z \not\subset \N \quad ; \quad
    \Z \not\subset \R^+$

    \item  \begin{itemize}
\item  $-7 \in \Z$.

\item On a $3{,}14 = \frac{314}{100} = \frac{314}{10^2}$. Donc $3{,}14 \in \D$.

\item On a $\frac{1}{3} = 0{,}333\ldots$. ~ Son écriture décimale est infinie et périodique , donc $\frac{1}{3} \notin \D$, 
Donc $\frac{1}{3} \in \Q$.

\item $\pi$ est un nombre irrationnel, il ne s'écrit pas comme quotient de deux entiers. Donc $\pi \in \R$ et $\pi \notin \Q$.

\item On a $\frac{12}{11} = 1{,}090909 ... $ ~ Son écriture décimale est infinie et périodique. Donc $\frac{12}{11} \in \Q$.

\item On a $\sqrt{1{,}44} = \sqrt{\frac{144}{100}} = \frac{12}{10} = 1{,}2$. Donc $\sqrt{1{,}44} \in \D$.

\item Puisque $\sqrt{3}$ est irrationnel, donc $\frac{-\sqrt{3}}{4}$ l'est aussi. \\
D'où $\frac{-\sqrt{3}}{4} \in \R$ et $\frac{-\sqrt{3}}{4} \notin \Q$.
\end{itemize}
\end{enumerate}
\end{Solution}

\begin{Solution}[ACTIVITÉ][2]
\begin{enumerate}
    \item On a :
        $(a+b)^3 = (a+b)^2(a+b) = \big(a^2 + 2ab + b^2\big)(a+b)$
    
    En distribuant :
    \[
     (a+b)^3   = a^3 + a^2 b + 2a^2 b + 2ab^2 + ab^2 + b^3 = a^3 + 3a^2 b + 3ab^2 + b^3
    \]

    \item On remplace $b$ par $-b$ :
    \[
        (a-b)^3 = a^3 + 3a^2(-b) + 3a(-b)^2 + (-b)^3 = a^3 - 3a^2 b + 3ab^2 - b^3
    \]

    \item   $(a-b)\big(a^2 + ab + b^2\big) = a^3 + a^2 b + ab^2 - a^2 b - ab^2 - b^3 = a^3 - b^3$
    

    \item     $(a+b)\big(a^2 - ab + b^2\big) = a^3 - a^2 b + ab^2 + a^2 b - ab^2 + b^3 = a^3 + b^3$

    \item On en déduit :
    \[
        a^3 - b^3 = (a-b)\big(a^2 + ab + b^2\big) \qquad ; \qquad a^3 + b^3 = (a+b)\big(a^2 - ab + b^2\big)
    \]
\end{enumerate}
\end{Solution}

\begin{Solution}[APPLICATION][6]
\begin{enumerate}
    \item \textbf{Montrons que $AB = 1$ :}
\begin{flalign*}
\text{On a :} \quad AB &= \sqrt{19 + 6\sqrt{10}} \times \sqrt{19 - 6\sqrt{10}} &\\
                       &= \sqrt{\big(19 + 6\sqrt{10}\big)\big(19 - 6\sqrt{10}\big)} &\\
                       &= \sqrt{19^2 - \big(6\sqrt{10}\big)^2} &\\
                       &= \sqrt{361 - 360} 
                       = \sqrt{1} = 1 
\end{flalign*}

    \item On pose $x = A + B$ et $y = A - B$.
    \begin{enumerate}
        \item \textbf{Calculons $x^2$ et $y^2$ :}
\begin{flalign*}
\text{On a :} \quad x^2 &= (A + B)^2 = A^2 + 2AB + B^2 &\\
                        &= \big(19 + 6\sqrt{10}\big) + 2 \times 1 + \big(19 - 6\sqrt{10}\big) 
                        = 40 \\[10pt]
\text{et :} \quad y^2 &= (A - B)^2 = A^2 - 2AB + B^2 &\\
                      &= \big(19 + 6\sqrt{10}\big) - 2 \times 1 + \big(19 - 6\sqrt{10}\big) 
                      = 36 
\end{flalign*}

        \item \textbf{Déduisons les écritures simplifiées de $x$ et $y$ :}

        $x = A + B$ est une somme de deux racines carrées,
        
        donc $x \geq 0$.
        et comme $x^2 = 40$, donc $x = \sqrt{40} = 2\sqrt{10}$.

        De plus $19 + 6\sqrt{10} > 19 - 6\sqrt{10}$, donc $A > B$ \\ d'où $y = A - B > 0$. \\
        Comme $y^2 = 36$, on a $y = \sqrt{36} = 6$.
    \end{enumerate}

    \item \textbf{Déterminons les écritures simplifiées de $A$ et $B$ :}

On a : $\begin{cases} x = 2\sqrt{10} \\ y = 6 \end{cases}$ ; donc : $\begin{cases} A + B = 2\sqrt{10} \quad (1) \\ A - B = 6 \quad (2) \end{cases}$

\smallskip
En additionnant les égalités $(1)$ et $(2)$ membre à membre, on obtient : $2A = 6 + 2\sqrt{10}$, d'où : $A = 3 + \sqrt{10}$.\\
Par ailleurs : $B = A - 6$. Donc : $B = -3 + \sqrt{10}$.

\end{enumerate}
\end{Solution}

                      &= (x - 2)\big(x^2 + 2x + 4 - 3x - 3\big) &\\
                      &= (x - 2)\big(x^2 - x + 1\big) &
\end{flalign*}

    \item \textbf{Développons et réduisons $E(x)$ :}
\begin{flalign*}
\text{On a :} \quad E(x) &= x^3 - 8 - 3(x - 2)(x + 1) &\\
                      &= x^3 - 8 - 3\big(x^2 - x - 2\big) &\\
                      &= x^3 - 8 - 3x^2 + 3x + 6 
                      = x^3 - 3x^2 + 3x - 2 &
\end{flalign*}
